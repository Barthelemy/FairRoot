79 Bool_t FairTutorialDet1::ProcessHits(FairVolume* vol)
80 {
81  /** This method is called from the MC stepping */
82 
83  cout << "In FairTutorialDet1::ProcessHits" << endl;
84  //Set parameters at entrance of volume. Reset ELoss.
85  if ( gMC->IsTrackEntering() ) {
86  fELoss = 0.;
87  fTime = gMC->TrackTime() * 1.0e09;
88  fLength = gMC->TrackLength();
89  gMC->TrackPosition(fPos);
90  gMC->TrackMomentum(fMom);
91  }
92 
93  // Sum energy loss for all steps in the active volume
94  fELoss += gMC->Edep();
95 
96  // Create FairTutorialDet1Point at exit of active volume
97  if ( gMC->IsTrackExiting() ||
98  gMC->IsTrackStop() ||
99  gMC->IsTrackDisappeared() ) {
100  fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
101  fVolumeID = vol->getMCid();
102  if (fELoss == 0. ) { return kFALSE; }
103  AddHit(fTrackID, fVolumeID, TVector3(fPos.X(), fPos.Y(), fPos.Z()),
104  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,
105  fELoss);
106 
107  // Increment number of tutorial det points in TParticle
108  FairStack* stack = (FairStack*) gMC->GetStack();
109  stack->AddPoint(kTutDet);
110  }
111 
112  return kTRUE;
113 }

150 void FairTutorialDet1::ConstructGeometry()
151 {
152  /** If you are using the standard ASCII input for the geometry
153  just copy this and use it for your detector, otherwise you can
154  implement here you own way of constructing the geometry. */
155 
156  FairGeoLoader* geoLoad = FairGeoLoader::Instance();
157  FairGeoInterface* geoFace = geoLoad->getGeoInterface();
158  FairTutorialDet1Geo* Geo = new FairTutorialDet1Geo();
159  Geo->setGeomFile(GetGeometryFileName());
160  geoFace->addGeoModule(Geo);
161 
162  Bool_t rc = geoFace->readSet(Geo);
163  if (rc) { Geo->create(geoLoad->getGeoBuilder()); }
164  TList* volList = Geo->getListOfVolumes();
165 
166  // store geo parameter
167  FairRun* fRun = FairRun::Instance();
168  FairRuntimeDb* rtdb= FairRun::Instance()->GetRuntimeDb();
169  FairTutorialDet1GeoPar* par=(FairTutorialDet1GeoPar*)(rtdb->getContainer("FairTutorialDet1GeoPar"));
170  TObjArray* fSensNodes = par->GetGeoSensitiveNodes();
171  TObjArray* fPassNodes = par->GetGeoPassiveNodes();
172 
173  TListIter iter(volList);
174  FairGeoNode* node = NULL;
175  FairGeoVolume* aVol=NULL;
176 
177  while( (node = (FairGeoNode*)iter.Next()) ) {
178  aVol = dynamic_cast<FairGeoVolume*> ( node );
179  if ( node->isSensitive() ) {
180  fSensNodes->AddLast( aVol );
181  } else {
182  fPassNodes->AddLast( aVol );
183  }
184  }
185  par->setChanged();
186  par->setInputVersion(fRun->GetRunId(),1);
187 
188  ProcessNodes ( volList );
189 }
