178 void FairDbTutParBin::fill(UInt_t rid)
179 {
180  // Get Reader Meta Class
181  fParam_Reader=GetParamReader();
182 
183  // Define a Context
184  ValTimeStamp ts(rid);
185  ValCondition context(FairDbDetector::kGfi,DataType::kData,ts);
186 
187  // Activate reading for this Context
188  fParam_Reader->Activate(context, GetVersion());
189 
190  // By default use the latest row entry
191  // (Other rows would correspond to outdated versions)
192  Int_t numRows = fParam_Reader->GetNumRows();
193  // Just use the latest row entry
194  if ( numRows > 1 ) { numRows = 1; }
195 
196  for (Int_t j = 0; j < numRows; ++j) {
197  FairDbTutParBin* cgd = (FairDbTutParBin*) fParam_Reader->GetRow(j);
198  if (!cgd) { continue; }
199  fTopPitch = cgd->GetTopPitch();
200  fTopAnchor = cgd->GetTopAnchor();
201  fTopNrFE = cgd->GetNrTopFE();
202  fFeType = cgd->GetFeType();
203  Int_t* i_arr = cgd->GetMyIArray();
204  Double_t* d_arr = cgd->GetMyDArray();
205  for(Int_t i=0; i<3; i++) { fMyIArray[i]=i_arr[i]; }
206  for(Int_t i=0; i<10; i++) { fMyDArray[i]=d_arr[i];}
207  fMyHisto = cgd->GetMyHisto();
208  }

214 void FairDbTutParBin::store(UInt_t rid)
215 {
216 
217 
218  DBLOG("FairDb", FairDbLog::kWarning) << "FairDbTutParBin Store() " << endl;
219 
220  // Boolean IO test variable
221  Bool_t fail= kFALSE;
222 
223  // Create a unique statement on choosen DB entry
224  auto_ptr<FairDbStatement> stmtDbn(fMultConn->CreateStatement(GetDbEntry()));
225  if ( ! stmtDbn.get() ) {
226  cout << "-E- FairDbTutParBin::Store() Cannot create a statement for Database_id: " << GetDbEntry()
227  << "\n Please check the FAIRDB_TSQL_* environment. Quitting ... " << endl;
228  exit(1);
229  }
230 
231  // Check if for this DB entry the table already exists.
232  // If not call the corresponding Table Definition Function
233  std::vector<std::string> sql_cmds;
234  TString atr(GetName());
235  atr.ToUpper();
236 
237  if (! fMultConn->GetConnection(GetDbEntry())->TableExists("FAIRDBTUTPARBIN") ) {
238  sql_cmds.push_back(FairDb::GetValDefinition("FAIRDBTUTPARBIN").Data());
239  sql_cmds.push_back(FairDbTutParBin::GetTableDefinition());
240  }
241 
242  // Packed SQL commands executed internally via SQL processor
243  std::vector<std::string>::iterator itr(sql_cmds.begin()), itrEnd(sql_cmds.end());
244  while( itr != itrEnd ) {
245  std::string& sql_cmd(*itr++);
246  stmtDbn->Commit(sql_cmd.c_str());
247  if ( stmtDbn->PrintExceptions() ) {
248  fail = true;
249  cout << "-E- FairDbTutParBin::Store() ******* Error Executing SQL commands *********** " << endl;
250  }
251 
252  }
253 
254  // Refresh list of tables in connected database
255  // for the choosen DB entry
256  fMultConn->GetConnection(GetDbEntry())->SetTableExists();
257 
258  // Writer Meta-Class Instance
259  fParam_Writer = GetParamWriter();
260 
261 
262  // Activate Writer Meta-Class with the proper
263  // Validity Time Interval (run_id encapsulated)
264 
265  // Writer Activate() arguments list
266  // <Arguments> <Type> <Comments>
267  //
268  // Validity Interval, ValInterval
269  // Composition , Int_t set via cont. factory
270  // Version , Int_t set via cont. factory
271  // DbEntry , Int_t set via cont. factory
272  // LogTitle , std::string set via cont. factory
273 
274  fParam_Writer->Activate(GetValInterval(rid),GetComboNo(), GetVersion(),GetDbEntry(),GetLogTitle());
275 
276  // Object to Table mapping
277  *fParam_Writer<< (*this);
278 
279  // Check for eventual IO problems
280  if ( !fParam_Writer->Close() ) {
281  fail = true;
282  cout << "-E- FairDbTutPar::Store() ******** Cannot do IO on class: " << GetName() << endl;
283  }
