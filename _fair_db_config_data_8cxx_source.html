63 void FairDbConfigData::Fill(FairDbResultPool& rs,
64  const FairDbValRecord* vrec)
65 {
66  // Skip First slot SeqNo in column.
67  UInt_t numParams = rs.NumCols()-1;
68  // Skip also ROW_ID if existing
69  if ( rs.HasRowCounter() ) { --numParams; }
70 
71  for (UInt_t iParam = 0; iParam < numParams; ++iParam ) {
72  Param* par = new Param;
73  par->Name = rs.CurColName();
74  par->Value = rs.CurColValue();
75  par->Type = rs.CurColFieldType();
76 
77  fParams.push_back(par);
78  rs.IncrementCurCol();
79  }
80 
81  fAggregateNo = vrec->GetAggregateNo ();
82 
83 }
