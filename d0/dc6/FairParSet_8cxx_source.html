48 Bool_t FairParSet::init()
49 {
50  // intitializes the container from an input in run time
51  // database. If this is not successful it is initialized from
52  // the second input. If this failes too, it returns an error.
53  // (calls internally the init function in the derived class)
54  FairRuntimeDb* rtdb=FairRuntimeDb::instance();
55  //FairRunAna* fRun =FairRunAna::Instance();
56 // cout << "-I- FairParSet::init() " << GetName() << endl;
57 
58  Bool_t allFound=kFALSE;
59  FairParIo* io=0;
60  if (rtdb) {
61  io=rtdb->getFirstInput();
62  if (io) { allFound=init(io); }
63  if (!allFound) {
64  io=rtdb->getSecondInput();
65  //cout << "-I FairParSet::init() 2 " << io << std::endl;
66  if (io) { allFound=init(io); }
67  } else { setInputVersion(-1,2); }
68  }
69  if (allFound) { return kTRUE; }
70  if (fLogger ) { fLogger->Error(MESSAGE_ORIGIN, "init() %s not initialized",GetName()); }
71  return kFALSE;
72 }
