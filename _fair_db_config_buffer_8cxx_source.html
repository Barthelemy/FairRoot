109 const FairDbConfigBuffer& FairDbConfigBuffer::operator>>(FairRegistry* reg)
110 {
111  if ( ! reg ) { return *this; }
112  Bool_t keysLocked = reg->KeysLocked();
113  Bool_t valuesLocked = reg->ValuesLocked();
114  reg->UnLockKeys();
115  reg->UnLockValues();
116  reg->Clear();
117 
118  if ( fCFSet ) {
119  UInt_t numParams = fCFSet->GetNumParams();
120 
121  if ( numParams == 3 && fCFSet->GetParamName(2) == "CONFIG_DATA" ) {
122  istringstream is(fCFSet->GetParamValue(2));
123  reg->ReadStream(is);
124  } else {
125  MAXDBLOG("FairDb",FairDbLog::kError,20) << "Attempting to fill FairRegistry from a table with "
126  << numParams << " columns (should be 3) using column named "
127  << fCFSet->GetParamName(2) << " (should be CONFIG_DATA)." << endl;
128  }
129  }
130  if ( keysLocked ) { reg->LockKeys(); }
131  if ( valuesLocked ) { reg->LockValues(); }
132  return *this;
133 }

159 Bool_t FairDbConfigBuffer::Write(UInt_t dbNo,
160  const std::string& logComment,
161  Bool_t localTest )
162 {
163 
164  if ( ! fCFSet ) {
165  MAXDBLOG("FairDb",FairDbLog::kError,20) << "No configuration data to write out." << endl;
166  return false;
167  }
168 
169  int requireGlobal = localTest ? -1 : 1;
170  if ( fVRec.GetAggregateNo() < 0 ) {
171  FairDbConnectionPool& cas = FairDbTableInterfaceStore::Instance().GetConnectionPool();
172  Int_t aggNo = cas.AllocateSeqNo(fVRec.GetTableInterface()->GetTableName(),requireGlobal,dbNo);
173  if ( aggNo <= FairDb::kMAXLOCALSEQNO && ! localTest ) {
174  DBLOG("FairDb",FairDbLog::kWarning) << "Cannot write out configuration data: no authorising entry in cascade." << endl;
175  return false;
176  }
177  fVRec.SetAggregateNo(aggNo);
178  fCFSetModified.SetAggregateNo(aggNo);
179  DBLOG("FairDb",FairDbLog::kInfo) << "Composite number: " << aggNo
180  << " allocated to entry " << fSoftwName
181  << "," << fConfigName
182  << " in table " << fVRec.GetTableInterface()->GetTableName() << endl;
183  }
184  FairDbWriter<FairDbConfigData> writer(fVRec,dbNo,logComment);
185  writer.SetRequireGlobalSeqno(requireGlobal);
186  writer.SetOverlayCreationDate();
187  writer << *fCFSet;
188  return writer.Close();
189 
190 }
