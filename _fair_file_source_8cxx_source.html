95 Bool_t FairFileSource::Init()
96 {
97  if(IsInitialized){
98  fLogger->Info(MESSAGE_ORIGIN, "FairFileSource already initialized");
99  return kTRUE;
100  }
101  if (!fInChain ) {
102  fInChain = new TChain("cbmsim", "/cbmroot");
103  fLogger->Info(MESSAGE_ORIGIN, "FairFileSource::Init() chain created");
104  }
105  fInChain->Add( fRootFile->GetName() );
106 
107  // Get the folder structure from file which describes the input tree.
108  // There are two different names possible, so check both.
109  fCbmroot= dynamic_cast <TFolder*> (fRootFile->Get("cbmroot"));
110  if(!fCbmroot) {
111  fCbmroot= dynamic_cast <TFolder*> (fRootFile->Get("cbmout"));
112  if(!fCbmroot) {
113  fCbmroot= gROOT->GetRootFolder()->AddFolder("cbmroot", "Main Folder");
114  } else {
115  fCbmroot->SetName("cbmroot");
116  }
117  }
118  // Get The list of branches from the input file and add it to the
119  // actual list of existing branches.
120  // Add this list of branches also to the map of input trees, which
121  // stores the information which branches belong to which input tree.
122  // There is at least one primary input tree, but there can be many
123  // additional friend trees.
124  // This information is needed to add new files to the correct friend
125  // tree. With this information it is also possible to check if the
126  // input files which are added to the input chain all have the same
127  // branch structure. Without this check it is possible to add trees
128  // with a different branch structure but the same tree name. ROOT
129  // probably only checks if the name of the tree is the same.
130  TList* list= dynamic_cast <TList*> (fRootFile->Get("BranchList"));
131  if(list==0)fLogger->Fatal(MESSAGE_ORIGIN, "No Branch list in input file");
132  TString chainName = fInputTitle;
133  TString ObjName;
134  fInputLevel.push_back(chainName);
135  fCheckInputBranches[chainName] = new std::list<TString>;
136  if(list) {
137  TObjString* Obj=0;
138  fLogger->Info(MESSAGE_ORIGIN, "Enteries in the list %i", list->GetEntries());
139  for(Int_t i =0; i< list->GetEntries(); i++) {
140  Obj=dynamic_cast <TObjString*> (list->At(i));
141  if(Obj!=0){
142  ObjName=Obj->GetString();
143  fLogger->Info(MESSAGE_ORIGIN, "Branch name %s", ObjName.Data());
144  fCheckInputBranches[chainName]->push_back(ObjName.Data());
145 
146  if(fBranchNameList->FindObject(ObjName.Data())==0) {
147  fBranchNameList->AddLast(Obj);
148  fBranchSeqId++;
149  }
150  }
151  }
152  }
153 
154  gROOT->GetListOfBrowsables()->Add(fCbmroot);
155  fListFolder->Add( fCbmroot );
156 
157  // Store the information about the unique runids in the input file
158  // together with the filename and the number of events for each runid
159  // this information is needed later to check if inconsitencies exist
160  // between the main input chain and any of the friend chains.
161 
162  // GetRunIdInfo(fInFile->GetName(), chainName);
163 
164  // Add all additional input files to the input chain and do a
165  // consitency check
166  std::list<TString>::const_iterator iter;
167  for(iter = fInputChainList.begin(); iter != fInputChainList.end(); iter++) {
168  // Store global gFile pointer for safety reasons.
169  // Set gFile to old value at the end of the routine.R
170  TFile* temp = gFile;
171 
172  // Temporarily open the input file to extract information which
173  // is needed to bring the friend trees in the correct order
174  TFile* inputFile = new TFile((*iter));
175  if (inputFile->IsZombie()) {
176  fLogger->Fatal(MESSAGE_ORIGIN, "Error opening the file %s which should be added to the input chain or as friend chain", (*iter).Data());
177  }
178 
179  // Check if the branchlist is the same as for the first input file.
180  Bool_t isOk = CompareBranchList(inputFile, chainName);
181  if ( !isOk ) {
182  fLogger->Fatal(MESSAGE_ORIGIN, "Branch structure of the input file %s and the file to be added %s are different.", fRootFile->GetName(), (*iter).Data());
183  return kFALSE;
184  }
185 
186  // Add the runid information for all files in the chain.
187  //GetRunIdInfo(inputFile->GetName(), chainName);
188  // Add the file to the input chain
189  fInChain->Add( (*iter) );
190 
191  // Close the temporarly file and restore the gFile pointer.
192  inputFile->Close();
193  gFile = temp;
194 
195  }
196  fNoOfEntries = fInChain->GetEntries();
197  fLogger->Info(MESSAGE_ORIGIN, "Entries in this Source %i------------", fNoOfEntries);
198  return kTRUE;
199

232 void FairFileSource::AddFriendsToChain()
233 {
234  // Loop over all Friend files and extract the type. The type is defined by
235  // the tree which is stored in the file. If there is already a chain of with
236  // this type of tree then the file will be added to this chain.
237  // If there is no such chain it will be created.
238  //
239  // Check if the order of runids and the event numbers per runid for all
240  // friend chains is the same as the one defined by the input chain.
241  // TODO: Should the order be corrected or should the execution be stopped.
242  // The order in the input tree defined by the order in which the files have
243  // been added. A file is defined by the runid.
244 
245  // In the old way it was needed sometimes to add a freind file more
246  // than once. This is not needed any longer, so we remove deuplicates
247  // from the list and display a warning.
248  std::list<TString> friendList;
249  std::list<TString>::iterator iter1;
250  for(iter1 = fFriendFileList.begin();
251  iter1 != fFriendFileList.end(); iter1++) {
252  if (find(friendList.begin(), friendList.end(), (*iter1))
253  == friendList.end()) {
254  friendList.push_back(*iter1);
255  }
256  }
257  // TODO: print a warning if it was neccessary to remove a filname from the
258  // list. This can be chacked by comparing the size of both list
259 
260  TFile* temp = gFile;
261 
262  Int_t friendType = 1;
263  // Loop over all files which have been added as friends
264  for(iter1 = friendList.begin();
265  iter1 != friendList.end(); iter1++) {
266  std::list<TString>::iterator iter;
267  TString inputLevel;
268  // Loop over all already defined input levels to check if this type
269  // of friend tree is already added.
270  // If this type of friend tree already exist add the file to the
271  // then already existing friend chain. If this type of friend tree
272  // does not exist already create a new friend chain and add the file.
273  Bool_t inputLevelFound = kFALSE;
274  TFile* inputFile;
275  for ( iter = fInputLevel.begin(); iter !=fInputLevel.end(); iter++ ) {
276  inputLevel = (*iter);
277 
278  inputFile = new TFile((*iter1));
279  if (inputFile->IsZombie()) {
280  fLogger->Fatal(MESSAGE_ORIGIN, "Error opening the file %s which should be added to the input chain or as friend chain", (*iter).Data());
281  }
282 
283  // Check if the branchlist is already stored in the map. If it is
284  // already stored add the file to the chain.
285  Bool_t isOk = CompareBranchList(inputFile, inputLevel);
286  if ( isOk ) {
287  inputLevelFound = kTRUE;
288  inputFile->Close();
289  continue;
290  }
291  inputFile->Close();
292  }
293  if (!inputLevelFound) {
294  inputLevel= Form("FriendTree_%i",friendType);
295  CreateNewFriendChain((*iter1), inputLevel);
296  friendType++;
297  }
298 
299  TChain* chain = (TChain*) fFriendTypeList[inputLevel];
300  chain->AddFile((*iter1), 1234567890, "cbmsim");
301  }
302  gFile=temp;
303 
304  // Check if all friend chains have the same runids and the same
305  // number of event numbers as the corresponding input chain
306  // CheckFriendChains();
307 
308  // Add all the friend chains which have been created to the
309  // main input chain.
310  std::map< TString, TChain* >::iterator mapIterator;
311  for (mapIterator = fFriendTypeList.begin();
312  mapIterator != fFriendTypeList.end(); mapIterator++ ) {
313 
314  TChain* chain = (TChain*)mapIterator->second;
315  fInChain->AddFriend(chain);
316  }
317 
318  // Print some output about the input structure
319  PrintFriendList();
320 
321 }

356 void FairFileSource::CheckFriendChains()
357 {
358  std::multimap< TString, std::multimap<TString,TArrayI> >::iterator it1;
359  std::multimap<TString,TArrayI> map1;
360 
361  // Get the structure from the input chain
362  it1=fRunIdInfoAll.find("InputChain");
363  map1 = it1->second;
364  std::vector<Int_t> runid;
365  std::vector<Int_t> events;
366  std::multimap<TString,TArrayI>::iterator it;
367  for ( it=map1.begin() ; it != map1.end(); it++ ) {
368  TArrayI bla = (*it).second;
369  runid.push_back(bla[0]);
370  events.push_back(bla[1]);
371  }
372 
373  // Now loop over all chains except the input chain and comapare the
374  // runids and event numbers.
375  // If there is a mismatch stop the execution.
376  Int_t errorFlag = 0;
377  TString inputLevel;
378  std::list<TString>::iterator listit;
379  for ( listit=fInputLevel.begin() ; listit != fInputLevel.end(); listit++ ) {
380  inputLevel = (*listit);
381  if ( !inputLevel.Contains("InputChain") ) {
382  it1=fRunIdInfoAll.find(inputLevel);
383  map1 = it1->second;
384  if ( runid.size() != map1.size()) {
385  errorFlag = 1;
386  // goto error_label;
387  break;
388  }
389  Int_t counter = 0;
390  for ( it=map1.begin() ; it != map1.end(); it++ ) {
391  TArrayI bla = (*it).second;
392  if ( (bla[0] != runid[counter]) || (bla[1] != events[counter]) ) {
393  errorFlag = 2;
394  // goto error_label;
395  break;
396  }
397  counter++;
398  }
399  if (errorFlag>0) {
400  break;
401  }
402  }
403  }
404 
405  // Use goto to leave double loop at once in case of error
406  // error_label:
407  if (errorFlag>0) {
408  fLogger->Error(MESSAGE_ORIGIN,"The input chain and the friend chain %s have a different structure:", inputLevel.Data());
409  if (errorFlag == 1) {
410  fLogger->Error(MESSAGE_ORIGIN,"The input chain has the following runids and event numbers:");
411  for ( Int_t i=0; i<runid.size(); i++) {
412  fLogger->Error(MESSAGE_ORIGIN," - Runid %i with %i events", runid[i], events[i]);
413  }
414  fLogger->Error(MESSAGE_ORIGIN,"The %s chain has the following runids and event numbers:", inputLevel.Data());
415  for ( it=map1.begin() ; it != map1.end(); it++ ) {
416  TArrayI bla = (*it).second;
417  fLogger->Error(MESSAGE_ORIGIN," - Runid %i with %i events", bla[0], bla[1]);
418  }
419  }
420  if (errorFlag == 2) {
421  Int_t counter = 0;
422  for ( it=map1.begin() ; it != map1.end(); it++ ) {
423  TArrayI bla = (*it).second;
424  fLogger->Error(MESSAGE_ORIGIN,"Runid Input Chain, %s chain: %i, %i", inputLevel.Data(), bla[0], runid[counter]);
425  fLogger->Error(MESSAGE_ORIGIN,"Event number Input Chain, %s chain: %i, %i", inputLevel.Data(), bla[1], events[counter]);
426  counter++;
427  }
428  }
429  fLogger->Fatal(MESSAGE_ORIGIN,"Event structure mismatch");
430  }
431 }

480 Bool_t FairFileSource::CompareBranchList(TFile* fileHandle, TString inputLevel)
481 {
482  // fill a set with the original branch structure
483  // This allows to use functions find and erase
484  std::set<TString> branches;
485  std::list<TString>::const_iterator iter;
486  for(iter = fCheckInputBranches[inputLevel]->begin();
487  iter != fCheckInputBranches[inputLevel]->end(); iter++) {
488  branches.insert(*iter);
489  }
490 
491  // To do so we have to loop over the branches in the file and to compare
492  // the branches in the file with the information stored in
493  // fCheckInputBranches["InputChain"]. If both lists are equal everything
494  // is okay
495 
496  // Get The list of branches from the input file one by one and compare
497  // it to the reference list of branches which is defined for this tree.
498  // If a branch with the same name is found, this branch is removed from
499  // the list. If in the end no branch is left in the list everything is
500  // fine.
501  set<TString>::iterator iter1;
502  TList* list= dynamic_cast <TList*> (fileHandle->Get("BranchList"));
503  if(list) {
504  TObjString* Obj=0;
505  for(Int_t i =0; i< list->GetEntries(); i++) {
506  Obj=dynamic_cast <TObjString*> (list->At(i));
507  iter1=branches.find(Obj->GetString().Data());
508  if (iter1 != branches.end() ) {
509  branches.erase (iter1);
510  } else {
511  // Not found is an error because branch structure is
512  // different. It is impossible to add to tree with a
513  // different branch structure
514  return kFALSE;
515  }
516  }
517  }
518  // If the size of branches is !=0 after removing all branches also in the
519  // reference list, this is also a sign that both branch list are not the
520  // same
521  if (branches.size() != 0 ) {
522  return kFALSE;
523  }
524 
525  return kTRUE;
526 }
