151 bool FairParTSQLIo::open()
152 {
153 
154  // Connection check ....
155  for (Int_t iEntry = 0; iEntry < fConnections->GetNumDb(); ++iEntry) {
156 
157  Bool_t fail= kFALSE;
158 
159  auto_ptr<FairDbStatement>
160  stmtDb(fConnections->CreateStatement(iEntry));
161  if ( ! stmtDb.get() ) {
162  DBLOG("FairDb",FairDbLog::kWarning) << "Cannot get a statement for DB entry# " << iEntry
163  << "\n ---> Please check the FAIRDB_TSQL_* environment. Transaction Failed ... " << endl;
164  return kFALSE;
165  }
166 
167 
168  // Now prepare Database Infrastructure for each DB entry
169  DBLOG("FairDb",FairDbLog::kInfo) << "-I- FairParTSQLIo:.open() checking DB entry# " << iEntry << endl;
170 
171  if (!fConnections->GetConnection(iEntry)->TableExists("FAIRDBGLOBALSEQNO") ) {
172 
173  std::vector<std::string> sql_cmds;
174  string sql1("CREATE TABLE ");
175  sql1 += "FAIRDBLOGENTRYVAL";
176  sql1 += "( SEQNO INT NOT NULL PRIMARY KEY AUTO_INCREMENT,";
177  sql1 += " TIMESTART DATETIME NOT NULL,";
178  sql1 += " TIMEEND DATETIME NOT NULL,";
179  sql1 += " DETID TINYINT,";
180  sql1 += " DATAID TINYINT,";
181  sql1 += " VERSION INT,";
182  sql1 += " COMPOSITEID INT,";
183  sql1 += " TIMEINCR DATETIME NOT NULL,";
184  sql1 += " TIMETRANS DATETIME NOT NULL);";
185 
186  string sql2("CREATE TABLE ");
187  sql2 += "FAIRDBLOGENTRY";
188  sql2 += "( SEQNO INT,";
189  sql2 += " ROW_ID INT,";
190  sql2 += " TABLENAME TEXT,";
191  sql2 += " MIN_SEQNO INT NOT NULL,";
192  sql2 += " MAX_SEQNO INT NOT NULL,";
193  sql2 += " N_SEQNO INT NOT NULL,";
194  sql2 += " USERNAME TEXT,";
195  sql2 += " PROCESS_ID TEXT,";
196  sql2 += " HOSTNAME TEXT,";
197  sql2 += " SERVERNAME TEXT,";
198  sql2 += " LOG_TITLE TEXT);";
199 
200  string sql3("CREATE TABLE ");
201  sql3 += "FAIRDBGLOBALSEQNO";
202  sql3 += "( TABLENAME CHAR(64) NOT NULL PRIMARY KEY,";
203  sql3 += " LASTUSEDSEQNO INT );";
204 
205  stringstream insert_sql;
206  insert_sql << "INSERT INTO FAIRDBGLOBALSEQNO VALUES ('*'," << GetGlobalSeqNoIn() << ");";
207 
208  sql_cmds.push_back(sql1);
209  sql_cmds.push_back(sql2);
210  sql_cmds.push_back(sql3);
211  sql_cmds.push_back(insert_sql.str());
212 
213 
214  // Now execute the assemble list of SQL commands.
215  auto_ptr<FairDbStatement> stmtDbn(fConnections->CreateStatement(iEntry));
216 
217  std::vector<std::string>::iterator itr(sql_cmds.begin()), itrEnd(sql_cmds.end());
218  while( itr != itrEnd ) {
219  std::string& sql_cmd(*itr++);
220  DBLOG("FairDb",FairDbLog::kInfo) << " Creating Global Meta Data at DB entry# "
221  << iEntry << " SQL:" << sql_cmd << endl;
222  stmtDbn->ExecuteUpdate(sql_cmd.c_str());
223  if ( stmtDbn->PrintExceptions() ) {
224  fail = true;
225  DBLOG("FairDb",FairDbLog::kFatal) << " ******* Error Creating Meta Data for DB Entry#"
226  << iEntry << endl;
227  }
228  }//! (sql_cmds)
229 
230  fConnections->SetAuthorisingEntry(iEntry);
231  fConnections->GetConnection(iEntry)->SetTableExists();
232 
233  DBLOG("FairDb",FairDbLog::kInfo) << " Meta-structure has been implemented ... continuing " << endl;
234 
235  } else {
236  // Some Meta-Structure in DB is already existing ... check that !
237  TSQLStatement* stmtTSQL = stmtDb->ExecuteQuery("select * from FAIRDBGLOBALSEQNO;");
238 
239  if ( stmtTSQL ) {
240  while ( stmtTSQL->NextResultRow() ) {
241  TString tableName = stmtTSQL->GetString(0);
242  int lastUsedSeqNo = stmtTSQL->GetInt(1);
243  int abase = lastUsedSeqNo / BAS_SEQNO;
244  SetGlobalSeqNoIn(abase);
245  if ( tableName == "*" && lastUsedSeqNo != GetGlobalSeqNoIn() ) {
246  DBLOG("FairDb",FairDbLog::kInfo) << "FAIRDBGLOBALSEQNO table error. Entry * = " << lastUsedSeqNo
247  << " but should be: " << GetGlobalSeqNoIn() << endl;
248  fail = kTRUE;
249  } else if ( tableName != "*" ) {
250  string tn3(tableName.Data(),3);
251  if ( lastUsedSeqNo < GetGlobalSeqNoIn() || lastUsedSeqNo > GetGlobalSeqNoOut() ) {
252  DBLOG("FairDb",FairDbLog::kInfo) << " FAIRDBGLOBALSEQNO table error. Entry " << tableName
253  << " but should start with FAIRDB \n"
254  << " and last used SEQNO: " << lastUsedSeqNo
255  << " should be in range [ " << GetGlobalSeqNoIn() << " .. " << GetGlobalSeqNoOut() << " ]" << endl;
256  fail = kTRUE;
257  }
258  }
259  }
260  }
261 
262  if ( fail ) {
263  DBLOG("FairDb",FairDbLog::kFatal)
264  << "-I- FairParTSQLIo::open() The FAIRDBGLOBALSEQNO table at DB entry# " << iEntry << " is already\n"
265  << " existing but not aligned in sequence nr. range [ " << GetGlobalSeqNoIn() << " .. " << GetGlobalSeqNoOut() << " ]"
266  << " quit rather than risk damage to this table ... \n" << endl;
267  return kFALSE;
268  }
269 
270  DBLOG("FairDb",FairDbLog::kInfo) << " Meta-structure for DB entry# " << iEntry
271  << " looks OK ... continuing" << endl;
272  }
273 
274  }//! entries
275 
276  // Activate SQL Io
277  FairRuntimeDb::instance()->activateParIo(this);
278  DBLOG("FairDb",FairDbLog::kInfo) << " connected with DB Entries# "
279  << fConnections->GetNumDb() << endl;
280  return kTRUE;
281 }
