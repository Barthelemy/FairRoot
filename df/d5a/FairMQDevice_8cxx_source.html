267 void FairMQDevice::LogSocketRates()
268 {
269  timestamp_t t0;
270  timestamp_t t1;
271 
272  timestamp_t msSinceLastLog;
273 
274  int numFilteredInputs = 0;
275  int numFilteredOutputs = 0;
276  vector<FairMQSocket*> filteredInputs;
277  vector<FairMQSocket*> filteredOutputs;
278 
279  int i = 0;
280  for (vector<FairMQSocket*>::iterator itr = fPayloadInputs->begin(); itr != fPayloadInputs->end(); itr++)
281  {
282  if (fLogInputRate.at(i) > 0)
283  {
284  filteredInputs.push_back((*itr));
285  ++numFilteredInputs;
286  }
287  ++i;
288  }
289 
290  i = 0;
291  for (vector<FairMQSocket*>::iterator itr = fPayloadOutputs->begin(); itr != fPayloadOutputs->end(); itr++)
292  {
293  if (fLogOutputRate.at(i) > 0)
294  {
295  filteredOutputs.push_back((*itr));
296  ++numFilteredOutputs;
297  }
298  ++i;
299  }
300 
301  vector<unsigned long> bytesIn(numFilteredInputs);
302  vector<unsigned long> msgIn(numFilteredInputs);
303  vector<unsigned long> bytesOut(numFilteredOutputs);
304  vector<unsigned long> msgOut(numFilteredOutputs);
305 
306  vector<unsigned long> bytesInNew(numFilteredInputs);
307  vector<unsigned long> msgInNew(numFilteredInputs);
308  vector<unsigned long> bytesOutNew(numFilteredOutputs);
309  vector<unsigned long> msgOutNew(numFilteredOutputs);
310 
311  vector<double> mbPerSecIn(numFilteredInputs);
312  vector<double> msgPerSecIn(numFilteredInputs);
313  vector<double> mbPerSecOut(numFilteredOutputs);
314  vector<double> msgPerSecOut(numFilteredOutputs);
315 
316  i = 0;
317  for (vector<FairMQSocket*>::iterator itr = filteredInputs.begin(); itr != filteredInputs.end(); itr++)
318  {
319  bytesIn.at(i) = (*itr)->GetBytesRx();
320  msgIn.at(i) = (*itr)->GetMessagesRx();
321  ++i;
322  }
323 
324  i = 0;
325  for (vector<FairMQSocket*>::iterator itr = filteredOutputs.begin(); itr != filteredOutputs.end(); itr++)
326  {
327  bytesOut.at(i) = (*itr)->GetBytesTx();
328  msgOut.at(i) = (*itr)->GetMessagesTx();
329  ++i;
330  }
331 
332  t0 = get_timestamp();
333 
334  while (fState == RUNNING)
335  {
336  try
337  {
338  t1 = get_timestamp();
339 
340  msSinceLastLog = (t1 - t0) / 1000.0L;
341 
342  i = 0;
343 
344  for (vector<FairMQSocket*>::iterator itr = filteredInputs.begin(); itr != filteredInputs.end(); itr++)
345  {
346  bytesInNew.at(i) = (*itr)->GetBytesRx();
347  mbPerSecIn.at(i) = ((double)(bytesInNew.at(i) - bytesIn.at(i)) / (1024. * 1024.)) / (double)msSinceLastLog * 1000.;
348  bytesIn.at(i) = bytesInNew.at(i);
349  msgInNew.at(i) = (*itr)->GetMessagesRx();
350  msgPerSecIn.at(i) = (double)(msgInNew.at(i) - msgIn.at(i)) / (double)msSinceLastLog * 1000.;
351  msgIn.at(i) = msgInNew.at(i);
352 
353  LOG(DEBUG) << "#" << fId << "." << (*itr)->GetId() << ": " << msgPerSecIn.at(i) << " msg/s, " << mbPerSecIn.at(i) << " MB/s";
354 
355  ++i;
356  }
357 
358  i = 0;
359 
360  for (vector<FairMQSocket*>::iterator itr = filteredOutputs.begin(); itr != filteredOutputs.end(); itr++)
361  {
362  bytesOutNew.at(i) = (*itr)->GetBytesTx();
363  mbPerSecOut.at(i) = ((double)(bytesOutNew.at(i) - bytesOut.at(i)) / (1024. * 1024.)) / (double)msSinceLastLog * 1000.;
364  bytesOut.at(i) = bytesOutNew.at(i);
365  msgOutNew.at(i) = (*itr)->GetMessagesTx();
366  msgPerSecOut.at(i) = (double)(msgOutNew.at(i) - msgOut.at(i)) / (double)msSinceLastLog * 1000.;
367  msgOut.at(i) = msgOutNew.at(i);
368 
369  LOG(DEBUG) << "#" << fId << "." << (*itr)->GetId() << ": " << msgPerSecOut.at(i) << " msg/s, " << mbPerSecOut.at(i) << " MB/s";
370 
371  ++i;
372  }
373 
374  t0 = t1;
375  boost::this_thread::sleep(boost::posix_time::milliseconds(fLogIntervalInMs));
376  }
377  catch (boost::thread_interrupted&)
378  {
379  LOG(INFO) << "FairMQDevice::LogSocketRates() interrupted";
380  break;
381  }
382  }
383 
384  LOG(INFO) << ">>>>>>> stopping FairMQDevice::LogSocketRates() <<<<<<<";
