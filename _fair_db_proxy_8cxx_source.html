69 void FairDbProxy::FindTimeLimits(const ValCondition& vc,
70  const FairDb::Version& task,
71  UInt_t dbNo,
72  ValTimeStamp earliestCreate,
73  ValTimeStamp& start,
74  ValTimeStamp& end) const
75 {
76  DBLOG("FairDb",FairDbLog::kInfo) << "FindTimeLimits for table: " << fTableName
77  << " context: " << vc
78  << " version: " << task
79  << " Earliest transaction date: " << earliestCreate
80  << " database: " << dbNo << endl;
81 
82 // Set the limits wide open
83  start = ValTimeStamp(0,0);
84  end = ValTimeStamp(0x7FFFFFFF,0);
85 
86 // Construct a Time Gate on the current date.
87 
88  const ValTimeStamp curVTS = vc.GetTimeStamp();
89  Int_t timeGate = FairDb::GetTimeWindow(this->GetTableName());
90  time_t vcSec = curVTS.GetSec() - timeGate;
91  ValTimeStamp startGate(vcSec,0);
92  vcSec += 2*timeGate;
93  ValTimeStamp endGate(vcSec,0);
94  string earliestCreateString(FairDb::MakeDateTimeString(earliestCreate));
95  string startGateString(FairDb::MakeDateTimeString(startGate));
96  string endGateString(FairDb::MakeDateTimeString(endGate));
97 
98 // Extract information for ValCondition.
99 
100  FairDbDetector::Detector_t detType(vc.GetDetector());
101  DataType::DataType_t simFlg(vc.GetDataType());
102 
103 // Use an auto_ptr to manage ownership of FairDbStatement and TSQLStatement
104  auto_ptr<FairDbStatement> stmtDb(fConnectionPool.CreateStatement(dbNo));
105 
106  for (int i_limit =1; i_limit <= 4; ++i_limit ) {
107  FairDbString sql("select ");
108  if ( i_limit == 1 ) sql << "min(TIMESTART) from " << fTableName
109  << "VAL where TIMESTART > '" << endGateString << "' ";
110  if ( i_limit == 2 ) sql << "min(TIMEEND) from " << fTableName
111  << "VAL where TIMEEND > '" << endGateString << "' ";
112  if ( i_limit == 3 ) sql << "max(TIMESTART) from " << fTableName
113  << "VAL where TIMESTART < '" << startGateString << "' ";
114  if ( i_limit == 4 ) sql << "max(TIMEEND) from " << fTableName
115  << "VAL where TIMEEND < '" << startGateString << "' ";
116  sql << " and DETID & " << static_cast<unsigned int>(detType)
117  << " and DATAID & " << static_cast<unsigned int>(simFlg)
118  << " and TIMEINCR >= '" << earliestCreateString << "'"
119  << " and VERSION = " << task;
120  DBLOG("FairDb",FairDbLog::kInfo) << " FindTimeLimits query id: " << i_limit
121  << " SQL:" <<sql.c_str() << endl;
122 
123  auto_ptr<TSQLStatement> stmt(stmtDb->ExecuteQuery(sql.c_str()));
124  stmtDb->PrintExceptions(0);
125 
126 // If the query returns data, convert to a time stamp and trim the limits
127  TString date;
128  if ( ! stmt.get() || ! stmt->NextResultRow() || stmt->IsNull(0) ) { continue; }
129  date = stmt->GetString(0);
130  if ( date.IsNull() ) { continue; }
131  ValTimeStamp ts(FairDb::MakeTimeStamp(date.Data()));
132  DBLOG("FairDb",FairDbLog::kInfo) << " FindTimeLimits query result: " << ts << endl;
133  if ( i_limit <= 2 && ts < end ) { end = ts; }
134  if ( i_limit >= 3 && ts > start ) { start = ts; }
135 
136  }
137 
138  DBLOG("FairDb",FairDbLog::kInfo)<< "FindTimeLimits for table: " << fTableName
139  << " found: " << start << " .. " << end << endl;
140

193 FairDbResultPool* FairDbProxy::QuerySeqNos(SeqList_t& seqNos,
194  UInt_t dbNo,
195  const string& sqlData,
196  const string& fillOpts) const
197 {
198  if ( seqNos.size() == 0 ) { return 0; }
199 
200  FairDbStopWatchManager::gStopWatchManager.RecMainQuery();
201  FairDbString sql;
202  sql << "select * from " << fTableName << " where ";
203 
204  if ( sqlData != "" ) { sql << "( "; }
205  Bool_t first = kTRUE;
206  SeqList_t::const_iterator itr1 = seqNos.begin();
207 
208  while ( itr1 != seqNos.end() ) {
209  UInt_t seq1 = *itr1;
210  UInt_t seq2 = seq1;
211  SeqList_t::const_iterator itr2 = itr1;
212  while ( itr2 != seqNos.end() && seq2 == *itr2 ) {
213  ++itr2;
214  ++seq2;
215  }
216  if ( first ) {
217  first = kFALSE;
218  } else {
219  sql << "or ";
220  }
221  if ( seq2 > seq1 + 1) {
222  sql << "SEQNO between " << seq1 << " and " << seq2-1 << ' ';
223  itr1 = itr2;
224  } else {
225  sql << "SEQNO = " << seq1 << ' ';
226  ++itr1;
227  }
228  }
229 
230  if ( sqlData != "" ) { sql << ") and " << "(" << sqlData << ")" << " "; }
231 
232  sql << "order by SEQNO";
233 
234  if ( FairDbUtils::OrderContextQuery() && fMetaData->HasRowCounter() ) {
235  sql << ",ROW_ID";
236  }
237 
238  DBLOG("FairDb",FairDbLog::kInfo) << "Database: " << dbNo
239  << " SeqIds query: " << sql.c_str() << endl;
240 
241 // Apply query and return result..
242  FairDbStatement* stmtDb = fConnectionPool.CreateStatement(dbNo);
243  return new FairDbResultPool(stmtDb,sql,fMetaData,fTableInterface,dbNo,fillOpts);

454 Bool_t FairDbProxy::ReplaceSeqNoOracle(UInt_t oldSeqNo,
455  UInt_t newSeqNo,
456  UInt_t dbNo) const
457 {
458 
459  FairDbResultPool* rsOld = QueryValidity(oldSeqNo,dbNo);
460  if ( rsOld && rsOld->IsBeforeFirst() ) { rsOld->FetchRow(); }
461  if ( ! rsOld || rsOld->IsExhausted() ) {
462  DBLOG("FairDb",FairDbLog::kWarning) << "Cannot renumber " << oldSeqNo
463  << " it does not exist" << endl;
464  delete rsOld;
465  return false;
466  }
467 
468 // Generate SQL to add new SeqNo in validity vable.
469 
470  FairDbString sql;
471  sql << "insert into " << fTableName << fValSuffix
472  << " values (" << newSeqNo;
473 
474  rsOld->IncrementCurCol();
475  UInt_t numCols =rsOld->NumCols();
476  for (UInt_t iCol = 2; iCol <= numCols; ++iCol) {
477  UInt_t concept = rsOld->CurColFieldType().GetConcept();
478  string delim = "";
479  if ( concept == FairDb::kString
480  || concept == FairDb::kDate
481  || concept == FairDb::kChar ) { delim = "\'"; }
482  sql << "," << delim << rsOld->CurColValue() << delim ;
483  rsOld->IncrementCurCol();
484  }
485  sql << ")";
486  DBLOG("FairDb",FairDbLog::kInfo) << "Database: " << dbNo
487  << " ReplaceSeqNo SQL (insert new VAL): " << sql.c_str() << endl;
488  delete rsOld;
489 
490 // Apply query.
491  auto_ptr<FairDbStatement> stmtDb(fConnectionPool.CreateStatement(dbNo));
492  if ( ! stmtDb.get() ) { return false; }
493  if ( ! stmtDb->ExecuteUpdate(sql.c_str()) || stmtDb->PrintExceptions() ) {
494  MAXDBLOG("FairDb",FairDbLog::kError,20)<< "SQL: " << sql.c_str()
495  << " Failed. " << endl;
496  return false;
497  }
498 
499 // Generate SQL to replace SeqNo in main table.
500  sql.GetString().erase();
501  sql << "update " << fTableName
502  << " set SEQNO = " << newSeqNo
503  << " where SEQNO = " << oldSeqNo << ";";
504 
505  DBLOG("FairDb",FairDbLog::kInfo) << "Database: " << dbNo
506  << " ReplaceSeqNo SQL (rename main entry): " << sql.c_str() << endl;
507 
508 // Apply query.
509  if ( ! stmtDb->ExecuteUpdate(sql.c_str()) || stmtDb->PrintExceptions() ) {
510  MAXDBLOG("FairDb",FairDbLog::kError,20) << "SQL: " << sql.c_str()
511  << " Failed. " << endl;
512  return false;
513  }
514 
515 // Generate SQL to remove old SeqNo in validity table.
516  sql.GetString().erase();
517  sql << "delete from " << fTableName << fValSuffix
518  << " where SEQNO = " << oldSeqNo << ";";
519 
520  DBLOG("FairDb",FairDbLog::kInfo) << "Database: " << dbNo
521  << " ReplaceSeqNo SQL (delete old VAL): " << sql.c_str() << endl;
522 
523 // Apply query.
524  if ( ! stmtDb->ExecuteUpdate(sql.c_str()) || stmtDb->PrintExceptions() ) {
525  MAXDBLOG("FairDb",FairDbLog::kError,20) << "SQL: " << sql.c_str()
526  << " Failed. " << endl;
527  return false;
528  }
529 
530  return true;

534 void FairDbProxy::CreateMetaData(FairDbTableMetaData& metaData) const
535 {
536 
537  const char* tableName = metaData.TableName().c_str();
538  DBLOG("FairDb",FairDbLog::kInfo) << "Create meta-data for table: " << tableName << endl;
539 
540  // Checking meta-data
541  for ( UInt_t dbNo = 0; dbNo < fConnectionPool.GetNumDb(); dbNo++ ) {
542  FairDbConnection* connection = fConnectionPool.GetConnection(dbNo);
543  if (!connection) {continue;}
544 
545  TSQLServer* server = connection->GetServer();
546  if (!server) {continue;}
547 
548  connection->Connect();
549  TSQLTableInfo* meta = server->GetTableInfo(tableName);
550  if ( ! meta ) {
551  connection->DisConnect();
552  continue;
553  }
554  DBLOG("FairDb",FairDbLog::kInfo) << "Meta-data query succeeded on db_id: " << dbNo << endl;
555 
556  // Clear out possible existing data
557  metaData.Clear();
558 
559  const TList* cols = meta->GetColumns();
560  TIter colItr(cols);
561  int col = 0;
562  while ( TSQLColumnInfo* colInfo = dynamic_cast<TSQLColumnInfo*>(colItr.Next()) ) {
563 
564  ++col;
565  string name(colInfo->GetName());
566  name = FairUtilString::ToUpper(name);
567  metaData.SetColName(name,col);
568 
569  FairDbFieldType fldType(colInfo->GetSQLType(),
570  colInfo->GetLength(),
571  colInfo->GetTypeName());
572 
573  // For now continue to check for unsigned
574  if ( !colInfo->IsSigned() ) { fldType.SetUnsigned(); }
575  metaData.SetColFieldType(fldType,col);
576 
577  metaData.SetColIsNullable(col,colInfo->IsNullable());
578 
579 
580  DBLOG("FairDb",FairDbLog::kInfo) << "Column: " << col << " | " << name << " | "
581  << " SQL_type: " << colInfo->GetSQLType() << " | "
582  << " SQL_type_def: " << colInfo->GetTypeName() << " | "
583  << " DB_data_type: " << fldType.AsString() << " | "
584  << " data_size: " << fldType.GetSize() << " | "
585  << " col_length: " << colInfo->GetLength() << endl;
586  }
587 
588  delete meta;
589  connection->DisConnect();
590  return;
591  }
