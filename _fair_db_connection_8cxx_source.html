160 Bool_t FairDbConnection::Open()
161 {
162 
163  this->ClearExceptionLog();
164 
165  if ( !this->IsClosed() ) { return kTRUE;}
166 
167 
168  if ( ! fUrl.IsValid() ) {
169  ostringstream oss;
170  oss << "Unable to open connection: URL '" << fUrl.GetUrl() << "' is invalid";
171  MAXDBLOG("FairDb",FairDbLog::kError,20) << oss.str() << endl;
172  fExceptionLog.AddEntry(oss.str());
173  return kFALSE;
174  }
175 
176  Int_t maxAttempt = fUrlValidated ? 100 : 20;
177  for (Int_t attempt = 1; attempt <= maxAttempt; ++attempt) {
178 
179  TString url_cstr = fUrl.GetUrl();
180  // Main Connection call
181  if (url_cstr.Contains("sqlite://")) {
182 
183  // Remove the tail from Url string
184  TString aFile(this->GetUrl());
185  aFile.Resize(aFile.Length()-1);
186  //cout << " SQLite file " << aFile << endl;
187 
188  fServer = TSQLServer::Connect(
189  aFile.Data(),
190  fUser.c_str(),
191  fPassword.c_str());
192 
193 
194  } else {
195  fServer = TSQLServer::Connect(
196  fUrl.GetUrl(),
197  fUser.c_str(),
198  fPassword.c_str());
199  }
200 
201  if ( ! fServer ) {
202  ostringstream oss;
203  oss << "-I- FairDbConnection::Open() Failing to open: " <<
204  fUrl.GetUrl() << " for user "
205  << fUser << " and password "
206  << fPassword << " (attempt "
207  << attempt << ")";
208 
209  fExceptionLog.AddEntry(oss.str());
210 
211  if ( attempt == 1 ) {
212  MAXDBLOG("FairDb",FairDbLog::kError,20) << oss.str() <<"retrying ..." << endl;
213  }
214  gSystem->Sleep(attempt*1000);
215  } else {
216  fServer->EnableErrorOutput(kTRUE);
217  if ( attempt > 1 ) {
218  DBLOG("FairDb",FairDbLog::kInfo) << "Connection opened on attempt "
219  << attempt << endl;
220  }
221  DBLOG("FairDb",FairDbLog::kInfo) << "Successfully opened connection to: "
222  << fUrl.GetUrl() << endl;
223 
224  // <DB> Check me about !!!
225  // ASCII database, populate it and make the connection permanent
226  // unless even ASCII DB connections are temporary.
227  return kTRUE;
228  }
229  }//! for on attempt
230 
231  MAXDBLOG("FairDb",FairDbLog::kError,20) << "Failed to open a connection to: "
232  << fUrl.GetUrl()
233  << " for user " << fUser << " and pwd " << fPassword << endl;
234 
235  return kFALSE;

306 void FairDbConnection::SetTableExists(const string& tableName)
307 {
308 
309  // cout << "-I- FairDbConnection::SetTableExists() ... " << tableName << endl;
310 
311  // Add name to list of existing tables
312  // (necessary when creating tables)
313  // Note: If tableName is null refresh list from the database.
314 
315  if ( tableName == "" ) {
316  TSQLStatement* stmt=NULL;
317  if ( fDbType == FairDb::kMySQL || fDbType == FairDb::kOracle ) {
318  // MYSQL or Oracle
319  stmt = CreatePreparedStatement("show tables");
320  } else if ( fDbType == FairDb::kPostgreSQL ) {
321  // PostGres using public only as schema
322  TString sql_cmd = "SELECT table_name FROM information_schema.tables WHERE table_schema = 'public' or table_schema = '" + fSchema + "';";
323  //cout << " sql_cmd -->" << sql_cmd << endl;
324  stmt = CreatePreparedStatement(sql_cmd.Data());
325  } else if ( fDbType == FairDb::kSQLite ) {
326  //SQLite
327  stmt = CreatePreparedStatement("SELECT name FROM sqlite_master WHERE type = 'table'");
328  } else {
329  DBLOG("FairDb",FairDbLog::kFatal)<< "Unknown SQL Server type:"<< fDbType << endl;
330  }
331 
332  if ( stmt ) {
333  if (stmt->Process()) {
334  stmt->StoreResult();
335  while (stmt->NextResultRow()) {
336  string tn(stmt->GetString(0));
337  std::transform(tn.begin(), tn.end(), tn.begin(), ::toupper);
338  //cout << " -I FairDbConnection::SetTableExists table name: " << tn << endl;
339  this->SetTableExists(tn);
340  }
341  }
342  delete stmt;
343  stmt = 0;
344  }
345  } else {
346  if ( ! this->TableExists(tableName) ) {
347  fExistingTableList += ",'";
348  fExistingTableList += tableName;
349  fExistingTableList += "'";
350  //cout << "-I- FairDbConnection::Table Registered ... " << tableName << endl;
351  }
352  }
353  // cout << "-I- FairDbConnection:: End of table exists ... " << endl;
354 }
