173 void FairDbLogEntry::SetReason(const string& reason)
174 {
175 
176  fReason = reason;
177 
178  if ( fReason.size() && fReason[0] == '@' ) {
179  string fileName(fReason,1);
180  fReason.clear();
181  ifstream reasonFile(fileName.c_str());
182  if ( ! reasonFile.is_open() ) {
183  DBLOG("FairDb",FairDbLog::kError) << "Cannot read \"Reason from File\" " << fileName << endl;
184  } else {
185  string line;
186  while ( ! reasonFile.eof() ) {
187  getline(reasonFile,line);
188  if ( line.substr(0,11) == "FIXUP-FILE:"
189  || line.substr(0,10) == "BROADCAST:" ) {
190  continue;
191  }
192  if ( fReason.size() ) { fReason += '\n'; }
193  fReason += line;
194  }
195  reasonFile.close();
196  }
197  }
198 }

225 Bool_t FairDbLogEntry::Write(UInt_t dbNo,
226  Int_t logSeqNo)
227 {
228 
229  if ( logSeqNo > 0 ) {
230  if ( fLogNumSeqNo == 0 ) {
231  fLogSeqNoMin = logSeqNo;
232  fLogSeqNoMax = logSeqNo;
233  }
234  if ( logSeqNo < fLogSeqNoMin ) { fLogSeqNoMin = logSeqNo; }
235  if ( logSeqNo > fLogSeqNoMax ) { fLogSeqNoMax = logSeqNo; }
236  ++fLogNumSeqNo;
237  }
238 
239  if ( ! this->HasReason() || fLogNumSeqNo == 0) {
240  DBLOG("FairDb",FairDbLog::kError) << "Cannot write LogEntry: no reason and/or no SEQNOs defined "
241  << endl;
242  return kFALSE;
243  }
244 
245  if ( fSeqNo != 0 && dbNo != fDbNo ) {
246  DBLOG("FairDb",FairDbLog::kError) << "Cannot write LogEntry: attempting to switch from database"
247  << fDbNo << " to " << dbNo << endl;
248  return kFALSE;
249  }
250 
251  fDbNo = dbNo;
252  FairDbTableInterface& tblProxy = FairDbTableInterfaceStore::Instance()
253  .GetTableInterface("FAIRDBLOGENTRY",this);
254  bool replace = true;
255 
256  // Allocate SEQNO before first write.
257  if ( fSeqNo == 0 ) {
258  replace = false;
259  int seqNo = tblProxy.GetConnectionPool().AllocateSeqNo("FAIRDBLOGENTRY",0,fDbNo);
260  if ( seqNo <= 0 ) {
261  DBLOG("FairDb",FairDbLog::kError) << "Cannot get Seq_Id for table FAIRDBLOGENTRY" << endl;
262  return kFALSE;
263  }
264  fSeqNo = seqNo;
265  }
266 
267  // Construct a FairDbValRecord.
268  ValInterval vr(fLogDetMask,fLogSimMask,fUpdateTime,fUpdateTime,"FairDbLogEntry");
269  FairDbValRecord vrec(vr,fLogVersion,-1,0);
270 
271  // Now build and output FairDbSqlValidityData.
272  FairDbSqlValidityData packet;
273  packet.Recreate("FAIRDBLOGENTRY",
274  vrec.GetValInterval(),
275  -1,
276  vrec.GetVersion(),
277  vrec.GetCreationDate());
278  packet.AddDataRow(tblProxy,0,*this);
279  packet.SetSeqNo(fSeqNo);
280  return packet.Store(fDbNo,replace);
281  ;
282 }
