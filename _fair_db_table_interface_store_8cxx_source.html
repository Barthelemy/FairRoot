140 void FairDbTableInterfaceStore::Config()
141 {
142 
143  FairRegistry& reg = this->GetConfig();
144 
145  fDataTypeAss.Set(reg);
146 
147  fRollbackTimes.Set(reg);
148 
149  this->ApplySqlCondition();
150 
151  const char* dir;
152  if ( reg.Get("Level2Cache",dir) ) {
153  TString tmp(dir);
154  gSystem->ExpandPathName(tmp);
155  if ( tmp.Contains("$" ) ) {
156  dir = "./";
157  DBLOG("FairDb",FairDbLog::kWarning)<< "Directory name expansion failed, using "
158  << dir << " instead" << endl;
159  } else {
160  dir = tmp.Data();
161  }
162 
163  FairDbBufferFile::SetWorkDir(dir);
164  DBLOG("FairDb",FairDbLog::kInfo) << "FairDbTableInterfaceStore: Setting L2 Cache to: " << dir << endl;
165  }
166 
167  int connectionsPermanent = 0;
168  if ( reg.Get("MakeConnectionsPermanent",connectionsPermanent) ) {
169  reg.RemoveKey("MakeConnectionsPermanent");
170  Int_t dbNo =fConnectionPool->GetNumDb();
171  if ( connectionsPermanent > 0 ) {
172  while ( --dbNo >= 0 ) { fConnectionPool->SetPermanent(dbNo); }
173  DBLOG("FairDb",FairDbLog::kInfo) << "Making all database connections permanent" << endl;
174  FairDbUtils::fAsciiDBConectionsTemporary = false;
175  } else {
176  while ( --dbNo >= 0 ) { fConnectionPool->SetPermanent(dbNo,false); }
177  DBLOG("FairDb",FairDbLog::kInfo) << "Forcing all connections, including ASCII DB, to be temporary" << endl;
178  FairDbUtils::fAsciiDBConectionsTemporary = true;
179  }
180  }
181 
182  // Check for request to order context queries and remove from the Registry.
183  int OrderContextQuery = 0;
184  if ( reg.Get("OrderContextQuery",OrderContextQuery) ) {
185  reg.RemoveKey("OrderContextQuery");
186  if ( OrderContextQuery ) {
187  FairDbUtils::fOrderContextQuery = true;
188  DBLOG("FairDb",FairDbLog::kInfo) << "Forcing ordering of all context queries" << endl;
189  }
190  }
191 
192  // Abort if Registry contains any unknown keys
193  const char* knownKeys[] = { "Level2Cache",
194  "Shutdown",
195  };
196  int numKnownKeys = sizeof(knownKeys)/sizeof(char*);
197  bool hasUnknownKeys = false;
198 
199  FairRegistry::FairRegistryKey keyItr(&this->GetConfig());
200  while ( const char* foundKey = keyItr() ) {
201  bool keyUnknown = true;
202  for (int keyNum = 0; keyNum < numKnownKeys; ++keyNum ) {
203  if ( ! strcmp(foundKey,knownKeys[keyNum]) ) { keyUnknown = false; }
204  }
205  if ( keyUnknown ) {
206  DBLOG("FairDb",FairDbLog::kInfo) << "Illegal registry item: " << foundKey << endl;
207  hasUnknownKeys = true;
208  }
209  }
210 
211  if ( hasUnknownKeys ) {
212  DBLOG("FairDb",FairDbLog::kInfo) << "Aborting due to illegal registry items." << endl;
213  abort();
214  }
215 }

218 FairDbTableInterface& FairDbTableInterfaceStore::GetTableInterface
219 (const std::string& tableNameReq,
220  const FairDbObjTableMap* tableRow)
221 {
222 
223 // Assume upper case name if cannot find request.
224  std::string tableName = tableNameReq;
225  std::string vldSuffix = "Validity";
226  if ( ! fConnectionPool->TableExists(tableName + vldSuffix ) ) {
227  tableName = FairUtilString::ToUpper(tableName);
228  vldSuffix = "VAL";
229  }
230  std::string proxyName = tableName;
231 
232  proxyName.append("::");
233  proxyName.append(tableRow->ClassName());
234  FairDbTableInterface* qpp = fTPmap[proxyName];
235  if ( ! qpp ) {
236  DBLOG("FairDb",FairDbLog::kInfo) <<" create a FairDbTableInterface " << tableRow << " proxyname# " << proxyName << endl;
237  qpp = new FairDbTableInterface(fConnectionPool,tableName,vldSuffix,tableRow);
238  this->ApplySqlCondition(qpp);
239  fTPmap[proxyName] = qpp;
240  }
241 
242  return *qpp;
243 
244 }
