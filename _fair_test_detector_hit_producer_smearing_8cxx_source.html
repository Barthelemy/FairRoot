86 // ---- Exec ----------------------------------------------------------
87 void FairTestDetectorHitProducerSmearing::Exec(Option_t* option)
88 {
89  fLogger->Debug(MESSAGE_ORIGIN, "Exec of FairTestDetectorHitProducerSmearing");
90 
91  fHitsArray->Delete();
92 
93  // fill the map
94  FairTestDetectorPoint* point = NULL;
95  FairTestDetectorHit* hit = NULL;
96  for (int iPoint = 0; iPoint < fPointsArray->GetEntriesFast(); iPoint++)
97  {
98  point = (FairTestDetectorPoint*)fPointsArray->At(iPoint);
99  if (!point)
100  {
101  continue;
102  }
103 
104  TVector3 position;
105  point->Position(position);
106 
107  TVector3 dposition;
108 
109  // simulate the detector response: smear position with gaussians
110  dposition.SetXYZ(0.05, 0.05, 0.);
111  position.SetXYZ(gRandom->Gaus(position.X(), 0.05), gRandom->Gaus(position.Y(), 0.05), position.Z());
112 
113  hit = new ((*fHitsArray)[iPoint]) FairTestDetectorHit(point->GetDetectorID(), iPoint, position, dposition);
114  }
115 }
