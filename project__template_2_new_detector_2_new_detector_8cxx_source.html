79 Bool_t NewDetector::ProcessHits(FairVolume* vol)
80 {
81  /** This method is called from the MC stepping */
82 
83  //Set parameters at entrance of volume. Reset ELoss.
84  if ( gMC->IsTrackEntering() ) {
85  fELoss = 0.;
86  fTime = gMC->TrackTime() * 1.0e09;
87  fLength = gMC->TrackLength();
88  gMC->TrackPosition(fPos);
89  gMC->TrackMomentum(fMom);
90  }
91 
92  // Sum energy loss for all steps in the active volume
93  fELoss += gMC->Edep();
94 
95  // Create NewDetectorPoint at exit of active volume
96  if ( gMC->IsTrackExiting() ||
97  gMC->IsTrackStop() ||
98  gMC->IsTrackDisappeared() ) {
99  fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
100  fVolumeID = vol->getMCid();
101  if (fELoss == 0. ) { return kFALSE; }
102  AddHit(fTrackID, fVolumeID, TVector3(fPos.X(), fPos.Y(), fPos.Z()),
103  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,
104  fELoss);
105 
106  // Increment number of NewDetector det points in TParticle
107  MyProjStack* stack = (MyProjStack*) gMC->GetStack();
108  stack->AddPoint(kNewDetector);
109  }
110 
111  return kTRUE;
112 }
