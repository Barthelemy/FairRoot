56 Bool_t Pythia8Generator::ReadEvent(FairPrimaryGenerator* cpg)
57 {
58  Int_t npart = 0;
59  while(npart == 0)
60  {
61  fPythia.next();
62  for(int i=0; i<fPythia.event.size(); i++)
63  {
64  if(fPythia.event[i].isFinal())
65  {
66 // only send HNL decay products to G4
67  if (fHNL != 0){
68  Int_t im = fPythia.event[i].mother1();
69  if (fPythia.event[im].id()==fHNL ){
70 // for the moment, hardcode 110m is maximum decay length
71  Double_t z = fPythia.event[i].zProd();
72  Double_t x = abs(fPythia.event[i].xProd());
73  Double_t y = abs(fPythia.event[i].yProd());
74  // cout<<"debug HNL decay pos "<<x<<" "<< y<<" "<< z <<endl;
75  if ( z < 11000. && z > 7000. && x<250. && y<250.) {
76  npart++;
77  }
78  }
79  }
80  else {npart++;}
81  };
82  };
83 // happens if a charm particle being produced which does decay without producing a HNL. Try another event.
84 // if (npart == 0){ fPythia.event.list();}
85  };
86 // cout<<"debug p8 event 0 " << fPythia.event[0].id()<< " "<< fPythia.event[1].id()<< " "<< fPythia.event[2].id()<< " "<< npart <<endl;
87  for(Int_t ii=0; ii<fPythia.event.size(); ii++){
88  if(fPythia.event[ii].isFinal())
89  {
90  Bool_t wanttracking=true;
91  if (fHNL != 0){
92  Int_t im = fPythia.event[ii].mother1();
93  if (fPythia.event[im].id() != fHNL) {wanttracking=false;}
94  }
95  if ( wanttracking ) {
96  Double_t z = fPythia.event[ii].zProd();
97  Double_t x = fPythia.event[ii].xProd();
98  Double_t y = fPythia.event[ii].yProd();
99  Double_t pz = fPythia.event[ii].pz();
100  Double_t px = fPythia.event[ii].px();
101  Double_t py = fPythia.event[ii].py();
102  cpg->AddTrack((Int_t)fPythia.event[ii].id(),px,py,pz,x,y,z,
103  (Int_t)fPythia.event[ii].mother1(),wanttracking);
104  // cout<<"debug p8->geant4 "<< wanttracking << " "<< ii << " " << fPythia.event[ii].id()<< " "<< fPythia.event[ii].mother1()<<" "<<x<<" "<< y<<" "<< z <<endl;
105  }

107 // Double_t vx, Double_t vy, Double_t vz, Int_t parent=-1,Bool_t wanttracking=true,Double_t e=-9e9);
108  };
109  if (fHNL != 0 && fPythia.event[ii].id() == fHNL){
110  Int_t im = (Int_t)fPythia.event[ii].mother1();
111  Double_t z = fPythia.event[ii].zProd();
