85 
86  fHitArray->Delete();
87 
88  if (FairRunAna::Instance()->IsTimeStamp())
89  {
90  fDigiArray = FairRootManager::Instance()->GetData("FairTestDetectorSortedDigi", fFunctor, FairRootManager::Instance()->GetEventTime() + 200);

92  }
93 
94  // fill the map
95 
96  for (int ipnt = 0; ipnt < fDigiArray->GetEntries(); ipnt++)
97  {
98  FairTestDetectorDigi* digi = (FairTestDetectorDigi*)fDigiArray->At(ipnt);
99  if (!digi)
100  {
101  std::cout << "-W- No digi!";
102  continue;
103  }
104 
105  // Double_t timestamp = digi->GetTimeStamp();
106  TVector3 pos(digi->GetX() + 0.5, digi->GetY() + 0.5, digi->GetZ() + 0.5);
107  TVector3 dpos(1 / TMath::Sqrt(12), 1 / TMath::Sqrt(12), 1 / TMath::Sqrt(12));
108 
109  FairTestDetectorHit* hit = new ((*fHitArray)[ipnt]) FairTestDetectorHit(-1, -1, pos, dpos);
110  hit->SetTimeStamp(digi->GetTimeStamp());
111  hit->SetTimeStampError(digi->GetTimeStampError());
112  hit->SetLink(FairLink("FairTestDetectorDigi", ipnt));
113 
114  fHitArray->Sort();
115  }
116 
117  fDigiArray->Delete();
118 }
