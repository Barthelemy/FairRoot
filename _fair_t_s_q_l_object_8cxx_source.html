106 /**
107  * Find the true type of the member named "mName".
108  *
109  *@param mName The name of the member to search for.
110  *
111  *@return The type of the member with the given name. If not found
112  * the string "UNKNOWN_OBJECT" is returned.
113  */
114 std::string* FairTSQLObject::GetMemberTypeName(char const* mName)
115 {
116  std::string* out;
117  // Init Member list.
118  InitMemList();
119 
120  // Find the object with the given name in the list
121  TObject* tmpOb = fMemberList->FindObject(mName);
122 
123  if(tmpOb) { //Found object, fetch type
124  TDataMember* dm = fCurCls->GetDataMember(tmpOb->GetName());
125  if(dm) {
126  out = new std::string(dm->GetTrueTypeName());
127  } else {
128  out = new std::string("UNKNOWN_OBJECT");
129  }
130  } else { // Did not find the object in the list
131  out = new std::string("UNKNOWN_OBJECT");
132  }
133  return out;
134 }

136 /**
137  * Find the true type of the member named "mName".
138  *
139  *@param mName The name of the member to search for.
140  *
141  *@return The type name of the member with the given name. If not
142  * found UNKNOWN_TYPE type is returned.
143  */
144 FairDBObjectMemberTypes FairTSQLObject::GetMemberType(std::string const& mName)
145 {
146  FairDBObjectMemberTypes curType;
147  std::string* tmpName = GetMemberTypeName(mName);
148 
149  if( (*tmpName) == "char") {
150  curType = CHAR;
151  } else if( (*tmpName) == "int") {
152  curType = INT;
153  } else if((*tmpName) == "unsigned int") {
154  curType = UINT;
155  } else if( (*tmpName) == "float") {
156  curType = FLOAT;
157  } else if( (*tmpName) == "double") {
158  curType = DOUBLE;
159  } else if( (*tmpName) == "TArrayI") {
160  curType = INT_ARRAY;
161  } else if( (*tmpName) == "TArrayI*" ) {
162  curType = INT_ARRAY_PTR;
163  } else if( (*tmpName) == "TArrayF" ) {
164  curType = FLOAT_ARRAY;
165  } else if ((*tmpName) == "TArrayF*") {
166  curType = FLOAT_ARRAY_PTR;
167  } else if( (*tmpName) == "TArrayD" ) {
168  curType = DOUBLE_ARRAY;
169  } else if ( (*tmpName) == "TArrayD*" ) {
170  curType = DOUBLE_ARRAY_PTR;
171  } else if ((*tmpName) == "UNKNOWN_OBJECT") {
172  curType = UNKNOWN_TYPE;
173  } else {
174  curType = COMPLEX_TYPE;
175  }
176  // Delete temporary string.
177  delete tmpName;
178  tmpName = 0;
179 
180  return curType;
181 }

295 /**
296  *@param mName The name of the member to search for.
297  *
298  *@return Handle to the actual data member of the class. If no
299  *getter function is defined or the member is not found, then the
300  *fields remain empty and the type is UNKNOWN_TYPE.
301  */
302 FairDBObjectMemberValue* FairTSQLObject::GetMember(std::string const& mName)
303 {
304  // Find object type
305  FairDBObjectMemberTypes type = GetMemberType(mName);
306 
307  // Create return value and set the type
308  FairDBObjectMemberValue* returnVal = new FairDBObjectMemberValue(type);
309  //(*returnVal).type = type;
310 
311  // Get the getter method.
312  TMethodCall* getMeth = GetDataMemberGetter(mName);
313 
314  if( (!getMeth) || (!getMeth->IsValid()) ) {
315  std::cerr << "<ERROR> Could not initialize getter method for "<< mName
316  << " Type is set to UNKNOWN_TYPE;" << std::endl;
317  (*returnVal).type = UNKNOWN_TYPE;
318  }
319 
320  std::string typeName = ", Type = ";
321  // Cast the object
322  char* tmpChar = 0;
323  long tmpLong = 0;
324  double tmpDouble = 0.00;
325 
326  switch( (*returnVal).type ) {
327  case CHAR:
328  typeName += "CHAR \n";
329  getMeth->Execute(this, "", &tmpChar);
330  (*returnVal).c_val = (*tmpChar);
331  break;
332  case INT:
333  typeName += "INT \n";
334  getMeth->Execute(this, "", tmpLong);
335  (*returnVal).i_val = static_cast<int>(tmpLong);
336  break;
337  case UINT:
338  typeName += "UNSIGNED INT \n";
339  getMeth->Execute(this, "", tmpLong);
340  (*returnVal).Ui_Val = tmpLong;
341  break;
342  case FLOAT:
343  typeName += "FLOAT \n";
344  getMeth->Execute(this, "", tmpDouble);
345  (*returnVal).f_val = static_cast<float>(tmpDouble);
346  break;
347  case DOUBLE:
348  typeName += "DOUBLE \n";
349  getMeth->Execute(this, "", tmpDouble);
350  (*returnVal).d_val = tmpDouble;
351  break;
352  case INT_ARRAY:
353  typeName += "INT_ARRAY \n";
354  getMeth->Execute(this, "", &tmpChar);
355  //(*returnVal).I_Ar_val = (TArrayI*)(tmpChar);
356  (*returnVal).I_Ar_val = reinterpret_cast<TArrayI*>(tmpChar);
357  break;
358  case INT_ARRAY_PTR:
359  typeName += "INT_ARRAY* \n";
360  getMeth->Execute(this, "", &tmpChar);
