333 void FairRunAnaProof::RunOneEvent(Long64_t entry)
334 {
335  if (fTimeStamps) {
336  RunTSBuffers();
337  } else {
338  UInt_t tmpId =0;
339  fRootManager->ReadEvent(entry);
340  tmpId = fEvtHeader->GetRunId();
341  if ( tmpId != fRunId ) {
342  fRunId = tmpId;
343  if ( !fStatic ) {
344  Reinit( fRunId );
345  fTask->ReInitTask();
346  }
347  }
348  fRootManager->StoreWriteoutBufferData(fRootManager->GetEventTime());
349 
350  fTask->ExecuteTask("");
351  fRootManager->Fill();
352  fRootManager->DeleteOldWriteoutBufferData();
353  fTask->FinishEvent();
354  }

360 {
361 // FairAnaSelector* proofSelector = new FairAnaSelector();
362 
363 
364  TChain* inChain = (TChain*)fRootManager->GetInChain();
365  TString par1File = "";
366  TString par2File = "";
367  if ( fRtdb->getFirstInput () ) {
368  par1File = fRtdb->getFirstInput ()->getFilename();
369  }
370  if ( fRtdb->getSecondInput() ) {
371  par2File = fRtdb->getSecondInput()->getFilename();

373 
374  TString outDir = (fOutputDirectory.Length()>1?fOutputDirectory.Data():gSystem->WorkingDirectory());

376  TString outFile = fRootManager->GetOutFile()->GetName();
377  fRootManager->CloseOutFile();

383  fProof->AddInput(new TNamed("FAIRRUNANA_fOutputDirectory",outDir.Data()));
384  fProof->AddInput(new TNamed("FAIRRUNANA_fOutputFileName",outFile.Data()));
385  fProof->AddInput(new TNamed("FAIRRUNANA_fParInput1FName",par1File.Data()));

387 
388  // uploading packages not needed, as the libraries are in the rootmap now
389  // fProof->ClearPackages();
390  // fProof->UploadPackage(fProofParName.Data());
391  // fProof->EnablePackage(fProofParName.Data());
392 
393  Int_t nofChainEntries = inChain->GetEntries();
394  LOG(INFO) << "FairRunAnaProof::RunOnProof(): The chain seems to have " << nofChainEntries << " entries." << FairLogger::endl;
395 
396  TObjArray* listOfFiles = inChain->GetListOfFiles();
397  LOG(INFO) << "FairRunAnaProof::RunOnProof(): There are " << listOfFiles->GetEntries() << " files in the chain." << FairLogger::endl;
398 
399  inChain->SetProof();
400 
401  Int_t nofEventsToAnalyze = NStop-NStart;
402 
403  if ( nofEventsToAnalyze <= 0 ) {
404  LOG(INFO) << "You requested to analyze events from " << NStart << " to " << NStop << " that is " << nofEventsToAnalyze << " events!!!" << FairLogger::endl;
405  nofEventsToAnalyze = nofChainEntries-NStart;
406  LOG(INFO) << "It will be changed to analyze all events from " << NStart << " to the end of chain (" << nofChainEntries << "), that is to analyze " << nofEventsToAnalyze << " events." << FairLogger::endl;
407  }
408 
409  LOG(INFO) << "FairRunAnaProof::RunOnProof(): Starting inChain->Process(\"FairAnaSelector\",\"\","
410  << nofEventsToAnalyze << "," << NStart << ")" << FairLogger::endl;
411  inChain->Process("FairAnaSelector","",nofEventsToAnalyze,NStart);
412  LOG(INFO) << "FairRunAnaProof::RunOnProof(): inChain->Process DONE" << FairLogger::endl;
413 
414  return;
415 }
