50 FairDbBufferFile::FairDbBufferFile(const char* fileName, Bool_t input )
51  :
52  fFile(0),

54  fHasErrors(kFALSE),
55  fArrayBuffer(0),

60  // fFileName = fileName;

62  Bool_t access = input ? fgReadAccess : fgWriteAccess;
63  if ( fgWorkDir == "" || ! access ) { fFileName = ""; }
64  else { fFileName = fgWorkDir + fFileName; }
65  }
66 
67  // Open the file.
68  ios_base::openmode mode = ios_base::in|ios_base::binary;
69  if ( ! input ) { mode = ios_base::out|ios_base::binary; }
70 
71  if ( fFileName == "" ) { fHasErrors = kTRUE; }
72  else {
73  fFile = new fstream(fFileName.c_str(),mode);
74  if ( ! fFile->is_open() || ! fFile->good() ) {
75  DBLOG("FairDb",FairDbLog::kWarning) << "Cannot open the file# " << fFileName
76  << "; all I/O will then fail. !!!" << endl;
77  fHasErrors = kTRUE;
78  }
79  }

205 FairDbBufferFile& FairDbBufferFile::operator >> (vector<FairDbObjTableMap*>& arr)
206 {
207  if ( ! this->CanRead() ) { return *this; }
208 
209  if ( arr.size() ) {
210  MAXDBLOG("FairDb",FairDbLog::kError,20) << "Attempting to read into non-empty array" << endl;
211  return *this;
212  }
213 
214 
215 
216  UInt_t marker = 0;
217  (*this) >> marker;
218  if ( marker != StartMarker ) {
219  MAXDBLOG("FairDb",FairDbLog::kError,20) << "Cannot find start of array marker" << endl;
220  this->Close();
221  this->CheckFileStatus();
222  return *this;
223  }
224 
225 
226  Int_t arrSize = 0;
227  (*this) >> arrSize;
228 
229  if ( arrSize ) {
230  Int_t objSize = 0;
231  string objName;
232  (*this) >> objName >> objSize;
233 
234 
235  TClass objClass(objName.c_str());
236  Int_t objSizefromRoot = objClass.Size();
237  void* obj = objClass.New();
238  void* vt = GetVTptr(obj);
239 
240  FairDbObjTableMap* tr = reinterpret_cast<FairDbObjTableMap*>(obj);
241  delete tr;
242 
243  DBLOG("FairDb",FairDbLog::kInfo)
244  << "Restoring array of " << arrSize << " "
245  << objName << " objects"
246  << " VTaddr " << hex << vt << dec
247  << " object size " << objSize << "(from file) "
248  << objSizefromRoot << "(from ROOT)"
249  << endl;
250 
251  if ( arrSize < 0 || objSize != objSizefromRoot ) {
252  MAXDBLOG("FairDb",FairDbLog::kError,20) << "Illegal array size ("<< arrSize
253  << ") or object size(" << objSize
254  << "," << objSizefromRoot << ")" << endl;
255  this->Close();
256  this->CheckFileStatus();
257  return *this;
258  }
259 
260 // Allocate buffer
261  delete[] fArrayBuffer;
262  Int_t buffSize = arrSize*objSize;
263  fArrayBuffer = new char[buffSize];
264  this->Read(fArrayBuffer,buffSize);
265 
266  char* elem = fArrayBuffer;
267  arr.reserve(arrSize);
268  for (int row = 0; row < arrSize; ++row ) {
269  SetVTptr(elem,vt);
270  arr.push_back(reinterpret_cast<FairDbObjTableMap*>(elem));
271  elem += objSize;
272  }
273 
274  }
275 
276  (*this) >> marker;
277  if ( marker != EndMarker ) {
278  MAXDBLOG("FairDb",FairDbLog::kError,20) << "Cannot find end of array marker" << endl;
279  this->Close();
280  this->CheckFileStatus();
281  }
282 
283  return *this;
284 
285 }

288 FairDbBufferFile& FairDbBufferFile::operator << (vector<FairDbObjTableMap*>& arr)
289 {
290 
291  if ( ! this->CanWrite() ) { return *this; }
292 
293  UInt_t marker = StartMarker;
294  (*this) << marker;
295  Int_t arrSize = arr.size();
296  (*this) << arrSize;
297 
298  if ( arrSize ) {
299  FairDbObjTableMap* obj = arr[0];
300  Int_t objSize = obj->IsA()->Size();
301  string objName = obj->ClassName();
302  (*this) << objName << objSize;
303  for (int row = 0; row < arrSize; ++row ) {
304  obj = arr[row];
305  const char* p = reinterpret_cast<const char*>(arr[row]);
306  this->Write(p,objSize);
307  }
308 
309  }
310 
311  marker = EndMarker;
312  (*this) << marker;
313 
314  return *this;
