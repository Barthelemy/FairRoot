79 // ----- Public method ReadEvent --------------------------------------
80 Bool_t FairPlutoGenerator::ReadEvent(FairPrimaryGenerator* primGen)
81 {
82 
83  // Check for input file
84  if ( ! fInputFile ) {
85  cout << "-E FairPlutoGenerator: Input file nor open!" << endl;
86  return kFALSE;
87  }
88 
89  // Check for number of events in input file
90  if ( iEvent > fInputTree->GetEntries() ) {
91  cout << "-E FairPlutoGenerator: No more events in input file!" << endl;
92  CloseInput();
93  return kFALSE;
94  }
95  TFile* g=gFile;
96  fInputFile->cd();
97  fInputTree->GetEntry(iEvent++);
98  g->cd();
99 
100  // Get PDG database
101  TDatabasePDG* dataBase = TDatabasePDG::Instance();
102 
103  // Get number of particle in TClonesrray
104  Int_t nParts = fParticles->GetEntriesFast();
105 
106  // Loop over particles in TClonesArray
107  for (Int_t iPart=0; iPart < nParts; iPart++) {
108  PParticle* part = (PParticle*) fParticles->At(iPart);
109  Int_t pdgType = dataBase->ConvertGeant3ToPdg( part->ID() );
110 
111  // Check if particle type is known to database
112  if ( ! pdgType ) {
113  cout << "-W FairPlutoGenerator: Unknown type " << part->ID()
114  << ", skipping particle." << endl;
115  continue;
116  }
117 
118  TLorentzVector mom = part->Vect4();
119  Double_t px = mom.Px();
120  Double_t py = mom.Py();
121  Double_t pz = mom.Pz();
122 
123  TVector3 vertex = part->getVertex();
124  Double_t vx = vertex.x();
125  Double_t vy = vertex.y();
126  Double_t vz = vertex.z();
127 
128  // Give track to PrimaryGenerator
129  primGen->AddTrack(pdgType, px, py, pz, vx, vy, vz);
130 
131  } // Loop over particle in event
132 
133  return kTRUE;
134 
135 }
