26 /**
27  * \class BinaryFunctor
28  * \brief Base class for all functors which are used in the FairTSBufferFunctional
29  * \see FairTSBufferFunctional
30  *
31  * The class is a base class to control which data is extracted by the FairTSBufferFunctional class for processing.
32  * The important method to overwrite is Call. It gets the actual data which is read in from the tree and a parameter.
33  * If the actual data is not anymore part of the data you want to have Call returns true to stop the reading of data.
34  * Otherwise it should return false.
35  * The method TimeOut is used to break the processing if for example always the same data is requested.
36  */

61  /**
62  * \parameter b: StopTime: All data older than StopTime is returned

64  bool Call(FairTimeStamp* a, double b) {
65  fRequestTime = b;
66  //std::cout << "StopTime: " << a->GetTimeStamp() << " > " << b << std::endl;

68  };
69 
70  bool TimeOut() {
71  if (fRequestTime != fOldTime) {
72  fOldTime = fRequestTime;
73  fSameTimeRequestCounter = 0;
74  //std::cout << "RequestedTime: " << fRequestTime << std::endl;
75  return false;
76  } else if (fRequestTime == fOldTime) {
77  std::cout << "-I- FairTSBufferFunctional StopTime has requested the same data as before: " << fRequestTime << std::endl;
78  fSameTimeRequestCounter++;
79  } else {
80  std::cout << "-E- FairTSBufferFunctional StopTime Functor has requested time " << fRequestTime << " smaller than old time " << fOldTime << std::endl;
81  return true;
82  }
83  if (fSameTimeRequestCounter > 9) {
84  return true;
85  } else { return false; }
86  }
87 
88  void ResetTimeOut() {fSameTimeRequestCounter = 0;}
89

133 /**
134  * \class FairTSBufferFunctional
135  * \brief A class to access time ordered data in a root branch
136  *
137  * In the constructor of the class one has to give the branch name of the data, the tree the data is stored in
138  * and a BinaryFunctor which contains the method how the data should be extracted. Several example functors already exists.
139  * To extract the data one has to call GetData with a parameter which fits to the selected functor.
140  * GetData returns a TClonesArray which contains the data.
141  *
142  *
143  * Be careful! The buffer runs through the time ordered data in one time direction only. This means that you cannot request data which is older than the
144  * data you have requested before.
145  *
146  * Addition: This is not true anymore. GetData(Double_t, Double_t) is able to get also data which is older but this only works if you request a fixed time
147  * via StopTime functor. For other functors the behavior is unpredictable.
148  *
149  * Created on: Feb 18, 201
150  * Author: stockman
151  */
