147 std::list<TString> FairDbStatement::TranslateSQL(const TString& sql)
148 {
149 // Translate MySQL specific to other dialects of SQL.
150 
151 
152  std::list<TString> sqlTransList;
153 
154  if ( fDbType == FairDb::kMySQL ) {
155  sqlTransList.push_back(sql);
156  return sqlTransList;
157 
158  } else {
159 
160  // Translation completion flag
161  Bool_t translated = kFALSE;
162  // Sql modification flag , further translation ongoing.
163  Bool_t modified = kFALSE;
164 
165  TString sqlTrans(sql);
166  sqlTrans.ToUpper();
167 
168 
169  if ( fDbType == FairDb::kOracle ) {
170 
171  // Format Date compatible with MySQL.
172  sqlTransList.push_back("ALTER SESSION SET NLS_DATE_FORMAT='YYYY-MM-DD hh24:mi:ss'");
173 
174  // Translate NOW()
175  if ( sqlTrans.Index("NOW()") != kNPOS ) {
176  sqlTrans.ReplaceAll("NOW()","SYSDATE");
177  modified = kTRUE;
178  }
179 
180  // Translate SHOW TABLES
181  if (sqlTrans == "SHOW TABLES" ) {
182  sqlTrans = "SELECT TABLE_NAME FROM ALL_TABLES";
183  sqlTransList.push_back(sqlTrans);
184  translated = kTRUE;
185  }
186 
187  TString sqlIncant;
188 
189  // Translate DROP TABLE commands.
190  Ssiz_t dropTableIndex = sqlTrans.Index("DROP TABLE",0,TString::kIgnoreCase );
191  if ( ! translated && dropTableIndex != kNPOS ) {
192  // Remove added "IF EXISTS"
193  sqlTrans.ReplaceAll(" IF EXISTS ", " ");
194 
195  //add the drop table command, then work on the synonym
196  sqlTransList.push_back(sqlTrans);
197 
198  // Extract the table name
199  Ssiz_t startIndex = dropTableIndex + 10;
200  while ( isspace(sqlTrans[startIndex]) ) { ++startIndex; }
201  Ssiz_t endIndex = startIndex + 1;
202  Ssiz_t endIndexMax = sqlTrans.Length();
203  while ( endIndex < endIndexMax
204  && (isalnum(sqlTrans[endIndex]) || sqlTrans[endIndex] == '_')
205  ) { ++endIndex; }
206  TString name(sqlTrans.Data()+startIndex,endIndex-startIndex);
207  translated = kTRUE;
208 
209  // Add extra ORACLE statmenent for a public synonym
210  TString sqlIncant1("DROP PUBLIC SYNONYM ");
211  sqlIncant1 += name ;
212  sqlTransList.push_back(sqlIncant1);
213  }
214 
215  Ssiz_t whereStart = sqlTrans.Index("INSERT INTO",0,TString::kIgnoreCase );
216  if ( whereStart == kNPOS ) { whereStart = 0; }
217  else {
218  Ssiz_t whereStartMax = sqlTrans.Length();
219  whereStart = sqlTrans.Index("(",whereStart,TString::kIgnoreCase );
220  if ( whereStart == kNPOS ) { whereStart = whereStartMax; }
221  else { ++whereStart; }
222  int level = 1;
223  while ( whereStart < whereStartMax && level ) {
224  char chr = sqlTrans[whereStart++];
225  if ( chr == '(' ) { ++level; }
226  if ( chr == ')' ) { --level; }
227  }
228  }
229 
230  Ssiz_t whereIndex = sqlTrans.Index(" WHERE ",whereStart,TString::kIgnoreCase );
231  if ( ! translated && whereIndex != kNPOS ) {
232  ++whereIndex;
233  Ssiz_t whereEnd = sqlTrans.Length();
234  std::string whereDelim[] = { "GROUP BY", "HAVING", "ORDER BY", "LIMIT" };
235  int numDelims = sizeof(whereDelim)/sizeof(string);
236  for (int iDelim = 0; iDelim < numDelims; ++iDelim ) {
237  const string& delimName = whereDelim[iDelim];
238  Ssiz_t delimIndex = sqlTrans.Index(delimName.c_str(), delimName.size(), whereIndex + 5, TString::kIgnoreCase );
239  if (delimIndex != kNPOS && delimIndex < whereEnd) { whereEnd = delimIndex; }
240  }
241 
242  // Translate all bitwise and expressions within the WHERE clause.
243  TString whereClause(sql.Data()+whereIndex,whereEnd-whereIndex);
244  // Convert \n to space so that tokenising works.
245  whereClause.ReplaceAll("\n"," ");
246 
247  std::vector<std::string> tokens;
248  FairUtilString::StringTok(tokens,whereClause.Data()," ");
249  int numTokens = tokens.size();
250  for (int ithToken = 1; ithToken < numTokens-1; ++ithToken ) {
251  if ( tokens[ithToken] == "&" ) {
252  string tmp("bitand(");
253  tmp += tokens[ithToken-1] + "," + tokens[ithToken+1] + ") != 0";
254  tokens[ithToken] = tmp;
255  tokens[ithToken-1] = "";
256  tokens[ithToken+1] = "";
257  }
258  }
259 
260  sqlTrans = sql(0,whereIndex);
261  for (int ithToken = 0; ithToken < numTokens; ++ithToken ) {
262  sqlTrans += " ";
263  sqlTrans += tokens[ithToken].c_str();
264  }
265  sqlTrans += " " + sql(whereEnd,999999);
266  modified = kTRUE;
267  }
268 
269  if ( ! translated && sqlTrans.Index("\\\'") != kNPOS ) {
270 
271  if ( not modified ) { sqlTrans = sql; }
272  sqlTrans.ReplaceAll("\\\'","\'\'");
273  modified = kTRUE;
274  }
275 
276  // PGSQL ------------------------------------------------------
277  } else if ( fDbType == FairDb::kPostgreSQL ) {
278  // Translate SHOW TABLES
279  if (sqlTrans == "SHOW TABLES" ) {
280  sqlTrans = "SELECT TABLE_NAME FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_SCHEMA = 'PUBLIC'";
281  sqlTransList.push_back(sqlTrans);
282  translated = kTRUE;
283  }
284 
285  // Boolean types
286  Ssiz_t whereStart = sqlTrans.Index("INSERT INTO",0,TString::kIgnoreCase );
287  if ( whereStart == kNPOS ) { whereStart = 0; }
288  else {
289  Ssiz_t whereStartMax = sqlTrans.Length();
290  whereStart = sqlTrans.Index("(",whereStart,TString::kIgnoreCase );
291  if ( whereStart == kNPOS ) { whereStart = whereStartMax; }
292  else { ++whereStart; }
293  int level = 1;
294  while ( whereStart < whereStartMax && level ) {
295  char chr = sqlTrans[whereStart++];
296  if ( chr == '(' ) { ++level; }
297  if ( chr == ')' ) { --level; }
298  }
299  }
300 
301  Ssiz_t whereIndex = sqlTrans.Index(" WHERE ",whereStart,TString::kIgnoreCase );
302  if ( ! translated && whereIndex != kNPOS ) {
303  ++whereIndex;
304  Ssiz_t whereEnd = sqlTrans.Length();
305  std::string whereDelim[] = { "GROUP BY", "HAVING", "ORDER BY", "LIMIT" };
306  int numDelims = sizeof(whereDelim)/sizeof(string);
307  for (int iDelim = 0; iDelim < numDelims; ++iDelim ) {
308  const string& delimName = whereDelim[iDelim];
309  Ssiz_t delimIndex = sqlTrans.Index(delimName.c_str(), delimName.size(),
310  whereIndex + 5, TString::kIgnoreCase );
311  if (delimIndex != kNPOS && delimIndex < whereEnd) { whereEnd = delimIndex; }
312  }
313 
314  // Translate all bitwise and expressions within the WHERE clause.
315  TString whereClause(sql.Data()+whereIndex,whereEnd-whereIndex);
316  // Convert \n to space so that tokenising works.
317  whereClause.ReplaceAll("\n"," ");
318 
319  std::vector<std::string> tokens;
320  FairUtilString::StringTok(tokens,whereClause.Data()," ");
321  int numTokens = tokens.size();
322  for (int ithToken = 1; ithToken < numTokens-1; ++ithToken ) {
323  if ( tokens[ithToken] == "&" ) {
324  string tmp("cast((");
325  tmp += tokens[ithToken-1] + " & " + tokens[ithToken+1] + ") As Boolean)";
326  tokens[ithToken] = tmp;
327  tokens[ithToken-1] = "";
328  tokens[ithToken+1] = "";
329  }
330  }
331 
332  sqlTrans = sql(0,whereIndex);
333  for (int ithToken = 0; ithToken < numTokens; ++ithToken ) {
334  sqlTrans += " ";
335  sqlTrans += tokens[ithToken].c_str();
336  }
337  sqlTrans += " " + sql(whereEnd,999999);
338  modified = kTRUE;
339 
340  }
341 
342  } //! PGSQL
343 
344  else if ( fDbType == FairDb::kSQLite ) {
345  // cout << "-I- FairDbStatement::TranslateSQL called for SQLite:" << sql << endl;
346 
347  }//! (SQLite)
348 
349 
350  // Translate CREATE TABLE commands using FairDbTableMetaData.
351  Ssiz_t createTableIndex = sqlTrans.Index("CREATE TABLE",0,TString::kIgnoreCase );
352 
353  if ( ! translated && createTableIndex != kNPOS ) {
354  FairDbTableMetaData tmd;
355  tmd.SetFromSql(sql.Data(), fDbType);
356  std::vector<std::string> sql_list;
357  FairUtilString::StringTok(sql_list,tmd.Sql(fDbType),";");
358  std::vector<std::string>::const_iterator itr(sql_list.begin()), itrEnd(sql_list.end());
359  while ( itr != itrEnd ) {
360  sqlTransList.push_back(*itr);
361  ++itr;
362  }
363  translated = kTRUE;
364  }
365 
366  // Finalizing translation
367  if ( modified && ! translated ) {
368  sqlTransList.push_back(sqlTrans);
369  translated = kTRUE;
370  }
371 
372  if ( translated ) {
373  DBLOG("FairDb",FairDbLog::kInfo)<< "TranslateSQL sql: " << sql << endl
374  << "translates to " << sqlTransList.size()
375  << " statements:- \n";
376  std::list<TString>::const_iterator itr(sqlTransList.begin()), itrEnd(sqlTransList.end());
377  while (itr != itrEnd) { DBLOG("FairDb",FairDbLog::kInfo) <<"TranslateSQL " << " " << *itr << endl; ++itr;}
378  } else {
379  sqlTransList.push_back(sql);
380  }
381 
382  return sqlTransList;
383  }
