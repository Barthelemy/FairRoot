162 // ----- Public method GenerateEvent -----------------------------------
163 Bool_t FairPrimaryGenerator::GenerateEvent(FairGenericStack *pStack) {
164  // Check for MCEventHeader
165  if (!fEvent) {
166  LOG(FATAL) << "No MCEventHeader branch!" << FairLogger::endl;
167  return kFALSE;
168  } else {
169 
170  // Initialise
171  fStack = pStack;
172  fNTracks = 0;
173  fEvent->Reset();
174 
175  // Create event vertex
176  MakeVertex();
177  fEvent->SetVertex(fVertex);
178 
179  // Create beam angle
180  // Here we only randomly generate two angles (anglex, angley)
181  // for the event and later on (in AddTrack())
182  // all our particles will be rotated accordingly.
183  if (fBeamAngle) {
184  MakeBeamAngle();
185  }
186 
187  // Create event plane
188  // Randomly generate an angle by which each track added (in AddTrack())
189  // to the event is rotated around the z-axis
190  if (fEventPlane) {
191  MakeEventPlane();
192  }
193 
194  // Call the ReadEvent methods from all registered generators
195  fListIter->Reset();
196  TObject *obj = 0;
197  FairGenerator *gen = 0;
198  while ((obj = fListIter->Next())) {
199  gen = dynamic_cast<FairGenerator *>(obj);
200  if (!gen) {
201  return kFALSE;
202  }
203  const char *genName = gen->GetName();
204  fMCIndexOffset = fNTracks; // number tracks before generator is called
205  Bool_t test = gen->ReadEvent(this);
206  if (!test) {
207  LOG(ERROR) << "ReadEvent failed for generator " << genName
208  << FairLogger::endl;
209  return kFALSE;
210  }
211  }
212 
213 
214  fTotPrim += fNTracks;
215  // Screen output
216 
217  // Set the event number if not set already by one of the dedicated generators
218  if (-1 == fEvent->GetEventID()) {
219  fEventNr++;
220  fEvent->SetEventID(fEventNr);
221  }
222  LOG(DEBUG) << "(Event " << fEvent->GetEventID() << ") " << fNTracks
223  << " primary tracks from vertex (" << fVertex.X() << ", "
224  << fVertex.Y() << ", " << fVertex.Z() << ") with beam angle ("
225  << fBeamAngleX << ", " << fBeamAngleY << ") " << FairLogger::endl;
226 
227  fEvent->SetNPrim(fNTracks);
228 
229  return kTRUE;
230  }
231 }

234 // ----- Public method AddTrack ----------------------------------------
235 void FairPrimaryGenerator::AddTrack(Int_t pdgid, Double_t px_raw,
236  Double_t py_raw, Double_t pz_raw,
237  Double_t vx, Double_t vy, Double_t vz,
238  Int_t parent, Bool_t wanttracking,
239  Double_t e, Double_t tof, Double_t weight) {
240 
241  // ---> Add event vertex to track vertex
242  vx += fVertex.X();
243  vy += fVertex.Y();
244  vz += fVertex.Z();
245 
246  // cout << "FairPrimaryGenerator::AddTrack +Z" << fVertex.Z() << " pdgid " <<
247  // pdgid << "?"<< wanttracking << endl;
248 
249  TVector3 mom(px_raw, py_raw, pz_raw);
250 
251  if (fEventPlane) {
252  // Rotate the track (event) by the randomly generated angle which is fixed
253  // for the complete event.
254  // The coordinate system is not changed by this rotation.
255  mom.RotateZ(fPhi);
256  }
257 
258  if (fBeamAngle) {
259  // Rotate the track (event) from the rotated beam direction system into
260  // the fixed global experiment coordinate system
261  mom.RotateUz(fBeamDirection.Unit());
262  }
263 
264  // ---> Convert K0 and AntiK0 into K0s and K0L
265  if (pdgid == 311 || pdgid == -311) {
266  Double_t test = gRandom->Uniform(0., 1.);
267  if (test >= 0.5) {
268  pdgid = 310;
269  } // K0S
270  else {
271  pdgid = 130;
272  } // K0L
273  }
274 
275  // ---> Check whether particle type is in PDG Database
276  TDatabasePDG *pdgBase = TDatabasePDG::Instance();
277  if (!pdgBase) {
278  Fatal("FairPrimaryGenerator", "No TDatabasePDG instantiated");
279  } else {
280  TParticlePDG *pdgPart = pdgBase->GetParticle(pdgid);
281  if (!pdgPart) {
282  if (e < 0) {
283  cerr << "\033[5m\033[31m -E FairPrimaryGenerator: PDG code " << pdgid
284  << " not found in database.\033[0m " << endl;
285  cerr << "\033[5m\033[31m -E FairPrimaryGenerator: Discarding particle "
286  "\033[0m " << endl;
287  cerr << "\033[5m\033[31m -E FairPrimaryGenerator: now MC Index is "
288  "corrupted \033[0m " << endl;
289  return;
290  } else {
291  cout << "\033[5m\033[31m -W FairPrimaryGenerator: PDG code " << pdgid
292  << " not found in database. This warning can be savely "
293  "ignored.\033[0m " << endl;
294  }
295  }
296  }
297  // ---> Get mass and calculate energy of particle
298  if (e < 0) {
299  Double_t mass = pdgBase->GetParticle(pdgid)->Mass();
300  e = TMath::Sqrt(mom.Mag2() + mass * mass);
301  } // else, use the value of e given to the function
302 
303  // ---> Set all other parameters required by PushTrack
304  Int_t doTracking = 0; // Go to tracking
305  if (fdoTracking && wanttracking) {
306  doTracking = 1;
307  }
308  Int_t dummyparent = -1; // Primary particle (now the value is -1 by default)
309  Double_t polx = 0.; // Polarisation
310  Double_t poly = 0.;
311  Double_t polz = 0.;
312  Int_t ntr = 0; // Track number; to be filled by the stack
313  Int_t status = 0; // Generation status
314 
315  if (parent != -1) {
316  parent += fMCIndexOffset;
317  } // correct for tracks which are in list before generator is called
318  // Add track to stack
319  fStack->PushTrack(doTracking, dummyparent, pdgid, mom.X(), mom.Y(), mom.Z(),
320  e, vx, vy, vz, tof, polx, poly, polz, kPPrimary, ntr,
321  weight, status, parent);
322  fNTracks++;
323 }

401 // ----- Private method MakeVertex -------------------------------------
402 void FairPrimaryGenerator::MakeVertex() {
403  Double_t vx = fBeamX0;
404  Double_t vy = fBeamY0;
405  Double_t vz;
406  if (1 == fNrTargets) {
407  vz = fTargetZ[0];
408  } else {
409  Int_t Target = (Int_t)gRandom->Uniform(fNrTargets);
410  vz = fTargetZ[Target];
411  }
412 
413  if (fSmearVertexZ)
414  vz = gRandom->Uniform(vz - fTargetDz / 2., vz + fTargetDz / 2.);
415 
416  if (fSmearGausVertexZ) {
417  vz = gRandom->Gaus(vz, fTargetDz);
418  }
419 
420  if (fSmearGausVertexXY) {
421  if (fBeamSigmaX != 0.) {
422  vx = gRandom->Gaus(fBeamX0, fBeamSigmaX);
423  }
424  if (fBeamSigmaY != 0.) {
425  vy = gRandom->Gaus(fBeamY0, fBeamSigmaY);
426  }
427  }
428 
429  if (fSmearVertexXY) {
430  if (fBeamSigmaX != 0.) {
431  vx = gRandom->Uniform(vx - fBeamSigmaX / 2., vx + fBeamSigmaX / 2.);
432  }
433  if (fBeamSigmaY != 0.) {
434  vy = gRandom->Uniform(vy - fBeamSigmaY / 2., vy + fBeamSigmaY / 2.);
435  }
436  }
437 
438  fVertex = TVector3(vx, vy, vz);
439 }
