50 FairIonGenerator::FairIonGenerator(const Char_t* ionName, Int_t mult,
51  Double_t px, Double_t py, Double_t pz,
52  Double_t vx, Double_t vy, Double_t vz)
53  :FairGenerator(),
54  fMult(mult),
55  fPx(px), fPy(py), fPz(pz),
56  fVx(vx), fVy(vy), fVz(vz),
57  fIon(NULL), fQ(0)
58 
59 {
60 
61  FairRunSim* fRun=FairRunSim::Instance();
62  TObjArray* UserIons=fRun->GetUserDefIons();
63  TObjArray* UserParticles=fRun->GetUserDefParticles();
64  FairParticle* part=0;
65  fIon =(FairIon*) UserIons->FindObject(ionName);
66  if(fIon) {
67  fgNIon++;
68  fMult = mult;
69  fPx = Double_t(fIon->GetA()) * px;
70  fPy = Double_t(fIon->GetA()) * py;
71  fPz = Double_t(fIon->GetA()) * pz;
72  fVx = vx;
73  fVy = vy;
74  fVz = vz;
75 
76  } else {
77  part= (FairParticle*)UserParticles->FindObject(ionName);
78  if(part) {
79  fgNIon++;
80  TParticle* particle=part->GetParticle();
81  fMult = mult;
82  fPx = Double_t(particle->GetMass()/0.92827231) * px;
83  fPy = Double_t(particle->GetMass()/0.92827231) * py;
84  fPz = Double_t(particle->GetMass()/0.92827231) * pz;
85  fVx = vx;
86  fVy = vy;
87  fVz = vz;
88  }
89  }
90  if(fIon==0 && part==0 ) {
91  cout << "-E- FairIonGenerator: Ion or Particle is not defined !" << endl;
92  Fatal("FairIonGenerator", "No FairRun instantised!");
93  }
94 
95 }
