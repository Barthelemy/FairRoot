306 Bool_t FairGeoNode::calcModuleTransform(FairGeoTransform& modTransform)
307 {
308  // Calculates the transformation relative to the detectors coordinate system
309  FairGeoNode* node=this;
310  modTransform=transform;
311  while (node&&!node->isModule()) {
312  FairGeoNode* pm=node->getMotherNode();
313  if (!pm) {
314  Error("calcModuleTransform","Mother volume of %s not found!",node->GetName());
315  return kFALSE;
316  }
317  modTransform.transFrom(pm->getTransform());
318  node=pm;
319  }
320  if (isTopNode()) { return kFALSE; }
321  else { return kTRUE; }
322 }

341 Int_t FairGeoNode::compare(FairGeoNode& rn)
342 {
343  // Compares the volume with the volume rn and prints the diagnose
344  // Returns kTRUE if the volume parameters are the same
345  Int_t diff[]= {0,0,0,0,0,0};
346  Int_t n=0;
347  cout<<fName<<'\t';
348  if (mother.CompareTo(rn.getMother())!=0) {
349  diff[0]=1;
350  n++;
351  }
352  if (medium&&rn.getMedium()) {
353  TString med=medium->GetName();
354  if (med.CompareTo(rn.getMedium()->GetName())!=0) {
355  diff[1]=1;
356  n++;
357  }
358  } else {
359  diff[1]=1;
360  n++;
361  }
362  if (shape.CompareTo(rn.getShape())!=0) {
363  diff[2]=1;
364  n++;
365  }
366  Int_t np=rn.getNumPoints();
367  if (points&&nPoints==np) {
368  for (Int_t i=0; i<np; i++) {
369  FairGeoVector v1=*(getPoint(i));
370  FairGeoVector v2=*(rn.getPoint(i));
371  if (fabs(v1(0)-v2(0))>=0.001||fabs(v1(1)-v2(1))>=0.001||fabs(v1(1)-v2(1))>=0.001) {
372  diff[3]=1;
373  n++;
374  break;
375  }
376  }
377  } else {
378  diff[3]=1;
379  n++;
380  }
381  FairGeoVector v1=transform.getTransVector();
382  FairGeoVector v2=rn.getTransform().getTransVector();
383  if (fabs(v1(0)-v2(0))>=0.001||fabs(v1(1)-v2(1))>=0.001||fabs(v1(1)-v2(1))>=0.001) {
384  diff[4]=1;
385  n++;
386  }
387  FairGeoRotation r1=transform.getRotMatrix();
388  FairGeoRotation r2=rn.getTransform().getRotMatrix();
389  for (Int_t i=0; i<9; i++) {
390  if (fabs(r1(i)-r2(i))>=1.e-7) {
391  diff[5]=1;
392  n++;
393  break;
394  }
395  }
396  if (n>0) {
397  for (Int_t i=0; i<6; i++) { cout<<" "<<diff[i]<<" "; }
398  cout<<'\n';
399  } else { cout<<" same\n"; }
400  return n;
401 }
