122 Int_t FairLmdSource::ReadEvent()
123 {
124  void* evtptr = &fxEvent;
125  void* buffptr = &fxBuffer;
126 
127  Int_t status = f_evt_get_event(fxInputChannel, (INTS4**)evtptr,(INTS4**) buffptr);
128  //Int_t fuEventCounter = fxEvent->l_count;
129  //Int_t fCurrentMbsEventNo = fuEventCounter;
130 
131  if(GETEVT__SUCCESS != status) {
132  if(fCurrentFile >= fFileNames->GetSize()) {
133  return 1;
134  }
135 
136  if(GETEVT__NOMORE == status) {
137  Close();
138  }
139 
140  TString name = ((TObjString*)fFileNames->At(fCurrentFile))->GetString();
141  if(! OpenNextFile(name)) {
142  return 1;
143  } else {
144  fCurrentFile += 1;
145  return ReadEvent();
146  }
147  }
148 
149  //Store Start Times
150  if (fCurrentEvent==0 )
151  Unpack((Int_t*)fxBuffer, sizeof(s_bufhe), -4, -4, -4, -4, -4);
152 
153 
154  // Decode event header
155  Bool_t result = Unpack((Int_t*)fxEvent, sizeof(s_ve10_1), -2, -2, -2, -2, -2);
156 
157  Int_t nrSubEvts = f_evt_get_subevent(fxEvent, 0, NULL, NULL, NULL);
158  Int_t sebuflength;
159  Short_t setype;
160  Short_t sesubtype;
161  Short_t seprocid;
162  Short_t sesubcrate;
163  Short_t secontrol;
164 
165  //if (fCurrentEvent%10000==0)
166  //cout << " -I- LMD_ANA: evt# " << fCurrentEvent << " n_subevt# " << nrSubEvts << " evt processed# " << fNEvent << " : " << fxEvent->l_count << endl;
167 
168 
169 // Int_t* SubEventDataPtr = new Int_t;
170  for(Int_t i = 1; i <= nrSubEvts; i++) {
171  void* SubEvtptr = &fxSubEvent;
172  void* EvtDataptr = &fxEventData;
173  Int_t* nrlongwords = new Int_t;
174  status = f_evt_get_subevent(fxEvent, i, (Int_t**)SubEvtptr, (Int_t**)EvtDataptr, nrlongwords);
175  if(status) {
176  return 1;
177  }
178  sebuflength = fxSubEvent->l_dlen;
179  setype = fxSubEvent->i_type;
180  sesubtype = fxSubEvent->i_subtype;
181  seprocid = fxSubEvent->i_procid;
182  sesubcrate = fxSubEvent->h_subcrate;
183  secontrol = fxSubEvent->h_control;
184  delete nrlongwords;
185 
186  if(Unpack(fxEventData, sebuflength,
187  setype, sesubtype,
188  seprocid, sesubcrate, secontrol)) {
189  result = kTRUE;
190  }
191  }
192 
193  // Increment evt counters.
194  fNEvent++;
195  fCurrentEvent++;
196 
197  if(! result)
198  {
199  return 2;
200  }
201 
202  return 0;
203 }
