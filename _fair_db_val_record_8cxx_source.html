303 void FairDbValRecord::Trim(const ValTimeStamp& queryTime,
304  const FairDbValRecord& other)
305 {
306 
307  if ( fAggregateNo != other.fAggregateNo || other.IsGap() ) { return; }
308 
309  if ( ! IsGap() ) { return; }
310 
311  ValTimeStamp start = fValInterval.GetTimeStart();
312  ValTimeStamp end = fValInterval.GetTimeEnd();
313  ValTimeStamp startOther = other.GetValInterval().GetTimeStart();
314  ValTimeStamp endOther = other.GetValInterval().GetTimeEnd();
315 
316  if ( startOther <= queryTime && endOther > queryTime ) {
317  if ( start < startOther ) { start = startOther; }
318  if ( end > endOther ) { end = endOther; }
319  *this = other;
320  SetTimeWindow(start,end);
321  }
322 
323  else {
324 
325  if ( endOther <= queryTime ) {
326  if ( start < endOther ) { SetTimeWindow(endOther,end); }
327  } else if ( startOther > queryTime ) {
328  if ( end > startOther ) { SetTimeWindow(start, startOther); }
329  }
330  }
