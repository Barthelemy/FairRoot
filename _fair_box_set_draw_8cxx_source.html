123  }
124 
125  //fList = (TClonesArray *)fManager->GetObject(GetName());
126  if (fVerbose > 1) {
127  std::cout << GetName() << " fList: " << fList->GetEntries() << std::endl;
128  }
129  for (Int_t i=0; i<fList->GetEntriesFast(); ++i) {
130  p=fList->At(i);

134  }
135  AddBoxes(fq, p, i);

137  gEve->AddElement(fq, fEventManager );
