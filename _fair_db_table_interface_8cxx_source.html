87 const FairDbResult* FairDbTableInterface::Query(const ValCondition& vc,
88  const FairDb::Version& task,
89  Bool_t findFullTimeWindow)
90 {
91 
92  DBLOG("FairDb",FairDbLog::kInfo) << "Query fulltimewindow " << findFullTimeWindow << endl;
93 // is info in cache?
94  if ( const FairDbResult* result = fCache->Search(vc,task)
95  ) { return result; }
96 
97  FairDbConnectionManager cm(fConnectionPool); //Stack object to hold connections
98 
99  UInt_t startGEL = FairDbExceptionLog::GetGELog().Size()+1;
100 
101  FairDbValRecordFactory factory(fDBProxy,vc,task,-1,findFullTimeWindow);
102 
103 
104  if ( factory.NonAggregated() ) {
105  DBLOG("FairDb",FairDbLog::kInfo) << " Non Composite Type " << endl;
106  FairDbValRecord effVRec = factory.GetValidityRec(0);
107  FairDbResult* result = const_cast<FairDbResult*>(Query(effVRec));
108  result->CaptureExceptionLog(startGEL);
109  return result;
110  }
111 
112 
113  DBLOG("FairDb",FairDbLog::kInfo) << "Composite Type " << endl;
114  if ( this->CanReadCache() ) {
115  DBLOG("FairDb",FairDbLog::kInfo) << "reading cache ... " << endl;
116  UInt_t numPresent = 0;
117  UInt_t numRequired = 0;
118  Int_t maxRow = factory.GetNumValidityRec() - 1;
119  for ( Int_t rowNo = 1; rowNo <= maxRow; ++rowNo ) {
120  const FairDbValRecord& vrec = factory.GetValidityRec(rowNo);
121  if ( fCache->Search(vrec) ) { ++numPresent; }
122  else if ( ! vrec.IsGap() ) { ++numRequired; }
123  }
124  if ( numRequired < numPresent )
125  DBLOG("FairDb",FairDbLog::kInfo) << "Skipping search of cache; already have "
126  << numPresent << " composites, and only require a further "
127  << numRequired << endl;
128  else { this->RestoreFromCache(factory); }
129  }
130 
131  DBLOG("FairDb",FairDbLog::kInfo) << "now do the results ---> " << endl;
132  FairDbResult* result = new FairDbResultCombo(fTableName,
133  fObjTableMap,
134  fCache,
135  &factory,
136  &fDBProxy);
137  result->CaptureExceptionLog(startGEL);
138 
139  fCache->Accept(result);
140  this->SaveToCache(factory.GetCacheName(),*result);
141  return result;
142

214 const FairDbResult* FairDbTableInterface::Query(const FairDbValRecord& vrec,
215  Bool_t canReuse )
216 {
217 
218  FairDbConnectionManager cm(fConnectionPool); //Stack object to hold connections
219 
220  UInt_t startGEL = FairDbExceptionLog::GetGELog().Size()+1;
221 
222  if ( canReuse ) {
223  FairDbValRecordFactory factory(vrec,this->GetTableName());
224  if ( this->RestoreFromCache(factory) ) {
225  const FairDbResult* res = fCache->Search(vrec);
226  if ( res ) { return res; }
227  }
228  }
229 
230  unsigned int seqNo = vrec.GetSeqNo();
231  FairDbResult* result = 0;
232 
233 
234  if ( ! seqNo ) {
235  result = new FairDbResultNonCombo(0,0,&vrec);
236  }
237 
238 
239  else if (vrec.GetTableInterface()->GetTableName() != GetTableName() ) {
240 
241  MAXDBLOG("FairDb",FairDbLog::kError,20)
242  << "Unable to satisfy FairDbValRecord keyed query:" << endl
243  << vrec
244  << " was filled by " << vrec.GetTableInterface()->GetTableName()
245  << " not by this FairDbTableInterface ("
246  << GetTableName() << ")" << endl;
247  result = new FairDbResultNonCombo(0,0,&vrec);
248  }
249 
250  else {
251  DBLOG("FairDb",FairDbLog::kInfo) << "Query(vc) SEQNO" << seqNo << endl;
252  FairDbResultPool* rs = fDBProxy.QuerySeqNo(seqNo,vrec.GetDbNo());
253  result = new FairDbResultNonCombo(rs,fObjTableMap,&vrec);
254  delete rs;
255  }
256 
257  result->CaptureExceptionLog(startGEL);
258 // Cache (memory + disk )
259  fCache->Accept(result);
260  if ( canReuse ) { this->SaveToCache(vrec.GetCacheName(),*result); }
261  else { result->SetCanReuse(kFALSE); }
262 
263  return result;
264 }

300 Bool_t FairDbTableInterface::RestoreFromCache(const FairDbValRecordFactory& factory)
301 {
302 
303  const string name(factory.GetCacheName());
304  DBLOG("FairDb",FairDbLog::kInfo) << "RestoreFromCache() Request to restore query result " << name
305  << endl;
306  if ( ! this->CanReadCache() ) { return kFALSE; }
307  string cacheFileName;
308  if ( name != ""
309  ) cacheFileName = this->GetTableName() + "_"
310  + this->GetRowName() + "_"
311  + name + ".db_cache";
312  FairDbBufferFile bf(cacheFileName.c_str());
313  if ( ! bf.IsOK() ) {
314  DBLOG("FairDb",FairDbLog::kInfo) << "Caching disabled or cannot open "
315  << bf.GetFileName() << endl;
316  return kFALSE;
317  }
318 
319  static bool warnOnce = true;
320  if ( warnOnce ) {
321  DBLOG("FairDb",FairDbLog::kWarning) << "\n\n\n"
322  << " WARNING: Reading from the Level 2 cache has been activated.\n"
323  << " ******* This mode should be used only for development and not for production !!!\n\n\n";
324  warnOnce = false;
325  }
326 
327  DBLOG("FairDb",FairDbLog::kInfo) << "Restoring query result from " << bf.GetFileName() << endl;
328  FairDbStopWatchManager::gStopWatchManager.RecMainQuery();
329 
330  FairDbResult* result = 0;
331  unsigned numRowsRest = 0;
332  unsigned numRowsIgn = 0;
333  UInt_t numNonAgg = 0;
334  bf >> numNonAgg;
335 
336  while ( numNonAgg-- ) {
337  if ( ! bf.IsOK() ) { break; }
338  if ( ! result ) { result = new FairDbResultNonCombo; }
339  bf >> *result;
340 
341  const FairDbValRecord& vrec = result->GetValidityRec();
342  UInt_t seqNo = vrec.GetSeqNo();
343  DBLOG("FairDb",FairDbLog::kInfo) << "Fix up cache FairDbValRecord, by replacing: " << vrec
344  << " with: " << factory.GetValidityRecFromSeqNo(seqNo) << endl;
345 
346  (const_cast<FairDbValRecord&>(vrec)) = factory.GetValidityRecFromSeqNo(seqNo);
347 
348  if ( ! fCache->Search(vrec) ) {
349  numRowsRest += result->GetNumRows();
350  fCache->Accept(result);
351  result = 0;
352  } else { numRowsIgn += result->GetNumRows(); }
353  }
354  DBLOG("FairDb",FairDbLog::kInfo) << " a total of " << numRowsRest << " were restored ("
355  << numRowsIgn << " ignored - already in memory)" << endl;
356 
357  delete result;
358  result = 0;
359 
360  return numRowsRest > 0;
361 }

363 Bool_t FairDbTableInterface::SaveToCache(const string& name, FairDbResult& res)
364 {
365 
366  DBLOG("FairDb",FairDbLog::kInfo) << "Request to save query result as " << name
367  << " data from DB? " << res.ResultsFromDb()
368  << " can be saved? " << res.CanSave() << endl;
369  if ( ! this->CanWriteCache() || ! res.ResultsFromDb() || ! res.CanSave() ) { return kFALSE; }
370 
371  string cacheFileName;
372  if ( name != ""
373  ) cacheFileName = this->GetTableName() + "_"
374  + this->GetRowName() + "_"
375  + name + ".db_cache";
376 
377  FairDbBufferFile bf(cacheFileName.c_str(),kFALSE);
378  if ( bf.IsOK() ) {
379  DBLOG("FairDb",FairDbLog::kInfo) << "Saving query result (" << res.GetNumRows()
380  << " rows) to " << bf.GetFileName() << endl;
381  FairDbStopWatchManager::gStopWatchManager.RecMainQuery();
382 
383  if ( dynamic_cast<FairDbResultNonCombo*>(&res) ) {
384  UInt_t numNonAgg = 1;
385  bf << numNonAgg;
386  }
387  bf << res;
388  return kTRUE;
389  }
390  DBLOG("FairDb",FairDbLog::kInfo) << "Caching disabled or cannot open "
391  << bf.GetFileName() << endl;
392  return kFALSE;
393 }
