108 void FairMultiLinkedData::AddLink(FairLink link, Bool_t bypass, Float_t mult)
109 {
110 
111  Float_t weight = link.GetWeight() * mult;
112  link.SetWeight(weight);
113  //std::cout << fVerbose << std::endl;
114 
115  FairRootManager* ioman = FairRootManager::Instance();
116 
117  if (ioman == 0) {
118  std::cout << "-W- no IOManager present!" << std::endl;
119  fPersistanceCheck = kFALSE;
120  }
121  if (fVerbose > 1) {
122  std::cout << "Add FairLink: " << link << std::endl;
123  }
124 
125  if (fPersistanceCheck == kFALSE ||
126  link.GetIndex() < 0 ||
127  ioman->CheckBranch(ioman->GetBranchName(link.GetType())) == 0) {
128  InsertLink(link);
129  InsertHistory(link);
130  return;
131  }
132 
133  if (bypass == kFALSE) {
134  if (fVerbose > 1) {

138  if (fVerbose > 1) {
139  std::cout << "BYPASS!" << std::endl;
140  }
141 // bypass = kTRUE;
142  }
143  }
144 
145  if (bypass == kTRUE) {
146  //FairRootManager* ioman = FairRootManager::Instance();
147  if (link.GetType() > ioman->GetBranchId("MCTrack")) {
148  TClonesArray* array = (TClonesArray*)ioman->GetObject(ioman->GetBranchName(link.GetType()));
149  if (fVerbose > 1) {
150  std::cout << "Entries in " << ioman->GetBranchName(link.GetType()) << " Array: " << array->GetEntries() << std::endl;
151  }
152  FairMultiLinkedData* links = (FairMultiLinkedData*)array->At(link.GetIndex());
153  if (fVerbose > 1) {
154  std::cout << "FairMultiLinkedData has " << links->GetNLinks() << " Entries: " << std::endl;
155  std::cout << *links << std::endl;

157  AddLinks(*links, mult);

159  } else {

170 void FairMultiLinkedData::InsertLink(FairLink link)
171 {
172  if (FairLinkManager::Instance()->IsIgnoreType(link.GetType())){
173  return;
174  }
175 
176  std::set<FairLink>::iterator it = fLinks.find(link);
177  if (it != fLinks.end()) {
178  FairLink myTempLink = *it;
179  myTempLink.AddWeight(link.GetWeight());
180  fLinks.erase(it);
181  fLinks.insert(myTempLink);
182  } else {
183  fLinks.insert(link);
184  }
185  return;
186 }

188 void FairMultiLinkedData::InsertHistory(FairLink link)
189 {
190  FairRootManager* ioman = FairRootManager::Instance();
191  FairMultiLinkedData* pointerToLinks = 0;
192 
193  if (link.GetType() < 0)
194  return;
195  if (link.GetType() == ioman->GetBranchId("MCTrack"))
196  return;
197 
198  if (link.GetIndex() < 0) { //if index is -1 then this is not a TClonesArray so only the Object is returned
199  FairMultiLinkedData_Interface* interface = (FairMultiLinkedData_Interface*) ioman->GetObject(ioman->GetBranchName(link.GetType()));
200  pointerToLinks = interface->GetPointerToData();
201  } else {
202  TClonesArray* dataArray = (TClonesArray*) ioman->GetObject(ioman->GetBranchName(link.GetType()));
203  if (link.GetIndex() < dataArray->GetEntriesFast()) {
204  FairMultiLinkedData_Interface* interface = (FairMultiLinkedData_Interface*) dataArray->At(link.GetIndex());
205  pointerToLinks = interface->GetPointerToData();
206 
207  }
208  }
209  if (pointerToLinks != 0){
210  std::set<FairLink> linkSet = pointerToLinks->GetLinks();
211  for (std::set<FairLink>::const_iterator iter = linkSet.begin(); iter!= linkSet.end(); iter++){
212  InsertLink(*iter);
213  }
214  }

256 TObject* FairMultiLinkedData::GetData(FairLink& myLink)
257 {

259  TString branchName = ioman->GetBranchName(myLink.GetType());
260  if (ioman->CheckBranch(branchName) > 0) {
261  TClonesArray* myArray = (TClonesArray*)ioman->GetObject(branchName);
262  if (myArray != 0) {
263  if (myArray->GetEntries() > myLink.GetIndex()) {
264  return myArray->At(myLink.GetIndex());
265  } else { std::cout << "-E- FairMultiLinkedData::GetData index out of bounds" << std::endl; }

267  } else { std::cout << "-E- FairMultiLinkedData::GetData Branch does not exist in Memory" << std::endl; }
