83 Bool_t FairTestDetector::ProcessHits(FairVolume* vol)
84 {
85  /** This method is called from the MC stepping */
86 
87  // Set parameters at entrance of volume. Reset ELoss.
88  if (gMC->IsTrackEntering())
89  {
90  fELoss = 0.;
91  fTime = gMC->TrackTime() * 1.0e09;
92  fLength = gMC->TrackLength();
93  gMC->TrackPosition(fPos);
94  gMC->TrackMomentum(fMom);
95  }
96 
97  // Sum energy loss for all steps in the active volume
98  fELoss += gMC->Edep();
99 
100  // Create FairTestDetectorPoint at exit of active volume
101  if (gMC->IsTrackExiting() || gMC->IsTrackStop() || gMC->IsTrackDisappeared())
102  {
103  fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
104  fVolumeID = vol->getMCid();
105  gMC->TrackPosition(fPosOut);
106  gMC->TrackMomentum(fMomOut);
107  if (fELoss == 0.)
108  {
109  return kFALSE;
110  }
111  AddHit(fTrackID,
112  fVolumeID,
113  TVector3(fPos.X(), fPos.Y(), fPos.Z()),
114  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()),
115  TVector3(fPosOut.X(), fPosOut.Y(), fPosOut.Z()),
116  TVector3(fMomOut.Px(), fMomOut.Py(), fMomOut.Pz()),
117  fTime,
118  fLength,
119  fELoss);
120 
121  // Increment number of FairTestDetector det points in TParticle
122  FairStack* stack = (FairStack*)gMC->GetStack();
123  stack->AddPoint(kTutDet);
124  }
125 
126  return kTRUE;
127 }

164 void FairTestDetector::ConstructGeometry()
165 {
166  /** If you are using the standard ASCII input for the geometry
167  just copy this and use it for your detector, otherwise you can
168  implement here you own way of constructing the geometry. */
169 
170  FairGeoLoader* geoLoad = FairGeoLoader::Instance();
171  FairGeoInterface* geoFace = geoLoad->getGeoInterface();
172  FairTestDetectorGeo* Geo = new FairTestDetectorGeo();
173  Geo->setGeomFile(GetGeometryFileName());
174  geoFace->addGeoModule(Geo);
175 
176  Bool_t rc = geoFace->readSet(Geo);
177  if (rc)
178  {
179  Geo->create(geoLoad->getGeoBuilder());
180  }
181  TList* volList = Geo->getListOfVolumes();
182 
183  // store geo parameter
184  FairRun* fRun = FairRun::Instance();
185  FairRuntimeDb* rtdb = FairRun::Instance()->GetRuntimeDb();
186  FairTestDetectorGeoPar* par = (FairTestDetectorGeoPar*)(rtdb->getContainer("FairTestDetectorGeoPar"));
187  TObjArray* fSensNodes = par->GetGeoSensitiveNodes();
188  TObjArray* fPassNodes = par->GetGeoPassiveNodes();
189 
190  TListIter iter(volList);
191  FairGeoNode* node = NULL;
192  FairGeoVolume* aVol = NULL;
193 
194  while ((node = (FairGeoNode*)iter.Next()))
195  {
196  aVol = dynamic_cast<FairGeoVolume*>(node);
197  if (node->isSensitive())
198  {
199  fSensNodes->AddLast(aVol);
200  }
201  else
202  {
203  fPassNodes->AddLast(aVol);
204  }
205  }
206  par->setChanged();
207  par->setInputVersion(fRun->GetRunId(), 1);
208 
209  ProcessNodes(volList);
210 }
