258 Bool_t FairDbSqlValidityData::Fill(std::ifstream& is)
259 {
260 
261  enum { kLOOKING_FOR_HEADER,
262  kLOOKING_FOR_TRAILER
263  } state = kLOOKING_FOR_HEADER;
264 
265  enum { kMAXTABLENAMELEN = FairDb::kMAXTABLENAMELEN,
266  kHEADER_TRAILER_MAX_LEN = kMAXTABLENAMELEN + 20
267  };
268 
269  string nameHead;
270  string nameTrail;
271  UInt_t seqNoHead = 0;
272  UInt_t seqNoTrail = 0;
273 
274  string line;
275  string msg;
276  string sql;
277  int lineNum = 0;
278 
279  this->Reset();
280 
281  while ( ! is.eof() ) {
282  getline(is,line);
283  ++lineNum;
284 
285  if (line.size() == 0 ) { continue; }
286 
287  if ( state == kLOOKING_FOR_HEADER ) {
288  if ( line.substr(0,5) == ">>>>>" ) {
289  if ( line.size() >= kHEADER_TRAILER_MAX_LEN ) {
290  Report("Bad header",lineNum,line);
291  continue;
292  }
293 
294  if ( line.find("Metadata") != string::npos ) {
295  getline(is,fSqlMySqlMetaVal);
296  ++lineNum;
297  getline(is,fSqlMySqlMetaMain);
298 
299  fgLastMetaHadRowCounter =
300  fSqlMySqlMetaMain.find("ROW_ID",fSqlMySqlMetaMain.find("(")) != string::npos;
301  ++lineNum;
302  getline(is,line);
303  ++lineNum;
304  if ( line.substr(0,5) != "<<<<<"
305  || line.find("Metadata") == string::npos ) {
306  Report("Bad metadata",lineNum,line);
307  continue;
308  }
309  getline(is,line);
310  ++lineNum;
311  if ( line.size() >= kHEADER_TRAILER_MAX_LEN ) {
312  Report("Bad header",lineNum,line);
313  continue;
314  }
315  }
316 
317  istringstream istr(line.substr(5));
318  istr.width(kMAXTABLENAMELEN);
319  istr >> nameHead >> seqNoHead;
320  if ( ! istr.eof() ) {
321  Report("Input error",lineNum,line);
322  continue;
323  }
324 
325  state = kLOOKING_FOR_TRAILER;
326  sql = "";
327  } else {
328  Report("Not header",lineNum,line);
329  }
330  }
331 
332 
333  else {
334  if ( line.substr(0,5) == "<<<<<" ) {
335  if ( line.size() >= kHEADER_TRAILER_MAX_LEN
336  ) { msg = "Bad trailer"; }
337 
338  else {
339  // Collect table name and SeqNo.
340  istringstream istr(line.substr(5));
341  istr.width(kMAXTABLENAMELEN);
342  istr >> nameTrail >> seqNoTrail;
343  if ( ! istr.eof() ) { msg = "Input error"; }
344 
345  else if ( nameTrail != nameHead
346  || seqNoHead != seqNoTrail ) {
347  msg = "Header/Trailer mismatch: Header: ";
348  msg += istr.str();
349  } else if ( GetNumSqlStmts() == 0
350  ) { msg = "No SQL statements between Header/Trailer"; }
351  else {
352 
353 
354  fSeqNo = seqNoHead;
355  fTableName = nameHead;
356 
357  string date = this->GetStmtValues(0)[7];
358 
359  date.erase(0,1);
360  date.erase(date.size()-1,1);
361  fCreationDate = FairDb::MakeTimeStamp(date);
362 
363  if ( fTableName == "FAIRDBSUBRUNSUMMARY" ) {
364  std::vector<std::string> ls;
365  FairUtilString::StringTok(ls,*fSqlStmts.begin(),",");
366 
367  if (ls.size() >= 7 && ls[6] != "-1") {
368  static bool warnOnce = true;
369  if ( warnOnce ) {
370 
371  DBLOG("FairDb",FairDbLog::kWarning) << "Setting composite no for FAIRDBRUNINFO = -1\n"
372  << " Expect this message once. FAIRDBRUNINFO needs to be "
373  << " fixed!" << endl;
374  warnOnce = false;
375  }
376  ls[6] = "-1";
377  sql = ls[0];
378  for (unsigned i=1; i<ls.size(); ++i ) { sql += "," + ls[i]; }
379  fSqlStmts.pop_front();
380  fSqlStmts.push_front(sql);
381  }
382  }
383  return kTRUE;
384 
385  }
386  }
387 
388  Report(msg.c_str(),lineNum,line);
389  state = kLOOKING_FOR_HEADER;
390  }
391 
392  else {
393  sql += line;
394  if ( sql[sql.size()-1] == ';') {
395  fSqlStmts.push_back(sql);
396  ++fNumStmts;
397  sql = "";
398  }
399  }
400  }
401 
402  }
403 
404  if ( state != kLOOKING_FOR_HEADER
405  ) { Report("Unexpected EOF",lineNum,"EOF"); }
406  this->Reset();
407  return kFALSE;
408 
409 }

436 Bool_t FairDbSqlValidityData::IsEqual(const FairDbSqlValidityData& that,
437  Bool_t log,
438  const Char_t* thisName,
439  const Char_t* thatName ) const
440 {
441 
442  if ( fSeqNo != that.fSeqNo
443  || fTableName != that.fTableName
444  || fNumStmts != that.fNumStmts ) {
445  if ( log ) {
446  DBLOG("FairDb",FairDbLog::kInfo) << "Conflict found:"
447  << " for : " << thisName << " , " << thatName << " :-\n"
448  << " SeqNo " << fSeqNo << "," << that.fSeqNo
449  << "\n TableName " << fTableName << "," << that.fTableName
450  << "\n Size " << fNumStmts << ","
451  << that.fNumStmts << endl;
452  }
453  return kFALSE;
454  }
455 
456  list<string>::const_iterator itrThisBegin = fSqlStmts.begin();
457  list<string>::const_iterator itrThisEnd = fSqlStmts.end();
458  list<string>::const_iterator itrThatBegin = that.fSqlStmts.begin();
459  list<string>::const_iterator itrThatEnd = that.fSqlStmts.end();
460 
461  list<string>::const_iterator itrThis = itrThisBegin;
462  list<string>::const_iterator itrThat = itrThatBegin;
463 
464  Bool_t isEqual = kTRUE;
465 
466 
467 
468  string strThis = (*itrThis).substr(0,(*itrThis).rfind(','));
469  string strThat = (*itrThat).substr(0,(*itrThat).rfind(','));
470  if ( strThis != strThat ) {
471  if ( ! log ) { return kFALSE; }
472  isEqual = kFALSE;
473 
474  DBLOG("FairDb",FairDbLog::kInfo) << "Difference on VALIDITY Record " << ":\n"
475  << " " << thisName << ": " << strThis << endl
476  << " " << thatName << ": " << strThat << endl;
477  }
478 
479 
480  ++itrThis;
481  ++itrThat;
482  while ( itrThis != itrThisEnd && (*itrThis) == (*itrThat) ) {
483  ++itrThis;
484  ++itrThat;
485  }
486 
487  if ( itrThis == itrThisEnd ) { return isEqual; }
488  itrThis = itrThisBegin;
489  itrThat = itrThatEnd;
490  ++itrThis;
491  --itrThat;
492 
493  while ( itrThis != itrThisEnd && (*itrThis) == (*itrThat) ) {
494  ++itrThis;
495  --itrThat;
496  }
497  if ( itrThis == itrThisEnd ) { return isEqual; }
498 
499 
500  typedef vector<const string*> shadow_list_t;
501  typedef shadow_list_t::iterator shadow_list_itr_t;
502 
503  shadow_list_t shadowThis;
504  shadow_list_t shadowThat;
505 
506  itrThis = itrThisBegin;
507  itrThat = itrThatBegin;
508  ++itrThis;
509  ++itrThat;
510 
511  while ( itrThis != itrThisEnd ) {
512  shadowThis.push_back(&(*itrThis));
513  shadowThat.push_back(&(*itrThat));
514  ++itrThis;
515  ++itrThat;
516  }
517 
518  shadow_list_itr_t shadowThisItr = shadowThis.begin();
519  shadow_list_itr_t shadowThisItrEnd = shadowThis.end();
520  shadow_list_itr_t shadowThatItr = shadowThat.begin();
521  shadow_list_itr_t shadowThatItrEnd = shadowThat.end();
522 
523  sort(shadowThisItr,shadowThisItrEnd,compStringPtrs);
524  sort(shadowThatItr,shadowThatItrEnd,compStringPtrs);
525 
526  while ( shadowThisItr != shadowThisItrEnd ) {
527  if ( (**shadowThisItr) != (**shadowThatItr) ) {
528  if ( ! log ) { return kFALSE; }
529  isEqual = kFALSE;
530  DBLOG("FairDb",FairDbLog::kInfo) << "Difference on data record "
531  << ":-\n"
532  << " " << thisName << ": " << **shadowThisItr << endl
533  << " " << thatName << ": " << **shadowThatItr << endl;
534  }
535  ++shadowThisItr;
536  ++shadowThatItr;
537  }
538 
539  return isEqual;
