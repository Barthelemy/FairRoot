129 std::vector<FairTimeStamp*> FairWriteoutBuffer::GetRemoveOldData(double time)
130 {
131  typedef std::multimap<double, FairTimeStamp*>::iterator DTMapIter;
132  std::vector<FairTimeStamp*> result;
133  for(DTMapIter it = fDeadTime_map.begin(); it != fDeadTime_map.lower_bound(time); it++) {
134  if (fVerbose > 1) {
135  std::cout << "-I- GetRemoveOldData: DeadTime: " << it->first << " Data: " << it->second << std::endl;
136  }
137  result.push_back(it->second);
138  EraseDataFromDataMap(it->second);
139  }
140  fDeadTime_map.erase(fDeadTime_map.begin(), fDeadTime_map.lower_bound(time));
141  return result;
142 }

168 void FairWriteoutBuffer::FillDataToDeadTimeMap(FairTimeStamp* data, double activeTime)
169 {
170  if (fActivateBuffering) {
171  typedef std::multimap<double, FairTimeStamp*>::iterator DTMapIter;
172  typedef std::map<FairTimeStamp, double>::iterator DataMapIter;
173 
174  double timeOfOldData = FindTimeForData(data);
175  // PrintDeadTimeMap();
176  if(timeOfOldData > -1) { //if an older active data object is already present
177  if (fVerbose > 1) {
178  std::cout << " OldData found! " << timeOfOldData << std::endl;
179  }
180  if (fVerbose > 1) {
181  std::cout << "New Data: " << activeTime << " : " << data << std::endl;
182  }
183  double currentdeadtime = timeOfOldData;
184  FairTimeStamp* oldData;
185  bool dataFound = false;
186 
187  // PrintDeadTimeMap();
188  for (DTMapIter it = fDeadTime_map.lower_bound(currentdeadtime); it != fDeadTime_map.upper_bound(currentdeadtime); it++) {
189  oldData = it->second;
190  if (fVerbose > 1) {
191  std::cout << "Check Data: " << it->first << " : " << oldData << std::endl;
192  }
193  if (oldData->equal(data)) {
194  if (fVerbose > 1) {
195  std::cout << " oldData == data " << std::endl;
196  }
197  if (fVerbose > 1) {
198  std::cout << it->first << " : " << it->second << std::endl;
199  }
200  fDeadTime_map.erase(it);
201  EraseDataFromDataMap(oldData);
202  dataFound = true;
203  break;
204  }
205  }
206 
207  if (dataFound == true) {
208  std::vector<std::pair<double, FairTimeStamp*> > modifiedData = Modify(std::pair<double, FairTimeStamp*>(currentdeadtime, oldData), std::pair<double, FairTimeStamp*>(-1, data));
209  for (int i = 0; i < modifiedData.size(); i++) {
210  FillDataToDeadTimeMap(modifiedData[i].second, modifiedData[i].first);
211  if (fVerbose > 1) {
212  std::cout << i << " :Modified Data: " << modifiedData[i].first << " : " << modifiedData[i].second << std::endl;
213  }
214  }
215  } else {
216  std::cout << "-E- FairWriteoutBuffer::FillDataToDeadTimeMap: old data present in dataMap but not in deadTimeMap!" << std::endl;
217  }
218 
219  } else {
220  if (fVerbose > 1) {
221  std::cout << "-I- FairWriteoutBuffer::FillDataToDeadTimeMap Data Inserted: " << activeTime << " : ";
222  data->Print();
223  std::cout << std::endl;
224  }
225  fDeadTime_map.insert(std::pair<double, FairTimeStamp*>(activeTime, data));
226  FillDataMap(data, activeTime);
227  }
228  } else {
229  AddNewDataToTClonesArray(data);
230  delete data;
231  }
232 }

235 void FairWriteoutBuffer::MoveDataFromStartTimeMapToDeadTimeMap(double time)
236 {
237  typedef std::multimap<double, std::pair<double, FairTimeStamp*> >::iterator startTimeMapIter;
238 
239  startTimeMapIter stopTime = fStartTime_map.lower_bound(time);
240  for (startTimeMapIter iter = fStartTime_map.begin(); iter != stopTime; iter++) {
241  std::pair<double, FairTimeStamp*> data = iter->second;
242  FillDataToDeadTimeMap(data.second, data.first);
243  }
244  fStartTime_map.erase(fStartTime_map.begin(), stopTime);
245 }
