137 Bool_t FairTutorialDet4::ProcessHits(FairVolume* vol)
138 {
139  /** This method is called from the MC stepping */
140 
141  //Set parameters at entrance of volume. Reset ELoss.
142  if ( gMC->IsTrackEntering() ) {
143  fELoss = 0.;
144  fTime = gMC->TrackTime() * 1.0e09;
145  fLength = gMC->TrackLength();
146  gMC->TrackPosition(fPos);
147  gMC->TrackMomentum(fMom);
148  }
149 
150  // Sum energy loss for all steps in the active volume
151  fELoss += gMC->Edep();
152 
153  // Create FairTutorialDet4Point at exit of active volume
154  if ( gMC->IsTrackExiting() ||
155  gMC->IsTrackStop() ||
156  gMC->IsTrackDisappeared() ) {
157  fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
158 // fVolumeID = vol->getMCid();
159  fVolumeID = fGeoHandler->GetUniqueDetectorId();
160  if (fELoss == 0. ) { return kFALSE; }
161 
162  // Save positions in local coordinate system, so transform the
163  // global coordinates into local ones.
164  Double_t master[3] = {fPos.X(), fPos.Y(), fPos.Z()};
165  Double_t local[3];
166 
167  gMC->Gmtod(master, local, 1);
168 
169 // AddHit(fTrackID, fVolumeID, TVector3(fPos.X(), fPos.Y(), fPos.Z()),
170  AddHit(fTrackID, fVolumeID, TVector3(local[0], local[1], local[2]),
171  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,
172  fELoss);
173 
174  // Increment number of tutorial det points in TParticle
175  FairStack* stack = (FairStack*) gMC->GetStack();
176  stack->AddPoint(kTutDet);
177  }
178 
179  return kTRUE;
180 }

240 void FairTutorialDet4::ConstructASCIIGeometry()
241 {
242  /** If you are using the standard ASCII input for the geometry
243  just copy this and use it for your detector, otherwise you can
244  implement here you own way of constructing the geometry. */
245 
246  FairGeoLoader* geoLoad = FairGeoLoader::Instance();
247  FairGeoInterface* geoFace = geoLoad->getGeoInterface();
248  FairTutorialDet4Geo* Geo = new FairTutorialDet4Geo();
249  LOG(DEBUG)<<"Read Geo file "<<GetGeometryFileName()<<FairLogger::endl;
250  Geo->setGeomFile(GetGeometryFileName());
251  geoFace->addGeoModule(Geo);
252 
253  Bool_t rc = geoFace->readSet(Geo);
254  if (rc) { Geo->create(geoLoad->getGeoBuilder()); }
255  TList* volList = Geo->getListOfVolumes();
256 
257  // store geo parameter
258  FairRun* fRun = FairRun::Instance();
259  FairRuntimeDb* rtdb= FairRun::Instance()->GetRuntimeDb();
260  FairTutorialDet4GeoPar* par=(FairTutorialDet4GeoPar*)(rtdb->getContainer("FairTutorialDet4GeoPar"));
261  TObjArray* fSensNodes = par->GetGeoSensitiveNodes();
262  TObjArray* fPassNodes = par->GetGeoPassiveNodes();
263 
264  TListIter iter(volList);
265  FairGeoNode* node = NULL;
266  FairGeoVolume* aVol=NULL;
267 
268  while( (node = (FairGeoNode*)iter.Next()) ) {
269  aVol = dynamic_cast<FairGeoVolume*> ( node );
270  if ( node->isSensitive() ) {
271  fSensNodes->AddLast( aVol );
272  } else {
273  fPassNodes->AddLast( aVol );
274  }
275  }
276  par->setChanged();
277  par->setInputVersion(fRun->GetRunId(),1);
278 
279  ProcessNodes ( volList );
280 }

284  // When saving the top volume of the geometry to a file the information
285  // about the TGeoPNEntries is not saved. So the misalign procedure
286  // which uses the TGeoPNEntry information can not be used.
287  // The code is here to demonstrate how to use this feature.
288 
289  if (fModifyGeometry) {

291 
292  TString detStr = "Tutorial4/det00";
293 
294  TGeoPNEntry* entry = gGeoManager->GetAlignableEntry(detStr.Data());
295  if (entry) {
296  LOG(INFO)<<"Misalign using symlinks."<<FairLogger::endl;
297  TGeoPhysicalNode* node = entry->GetPhysicalNode();
298 // LOG(INFO)<<"Nr of alignable objects: "<<gGeoManager->GetNAlignable()<<FairLogger::endl;
299  ModifyGeometryBySymlink();

301  LOG(INFO)<<"Misalign using full path."<<FairLogger::endl;
