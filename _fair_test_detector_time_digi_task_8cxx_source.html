75 
76  // fDigiArray->Delete();

78  // fill the map

82  for (int ipnt = 0; ipnt < fPointArray->GetEntries(); ipnt++)
83  {
84  FairTestDetectorPoint* point = (FairTestDetectorPoint*)fPointArray->At(ipnt);
85  if (!point)

87  continue;
88  }
89 
90  Int_t xPad = CalcPad(point->GetX(), point->GetXOut());
91  Int_t yPad = CalcPad(point->GetY(), point->GetYOut());
92  Int_t zPad = CalcPad(point->GetZ(), point->GetZOut());
93 
94  Double_t timestamp = CalcTimeStamp(point->GetTime());
95 
96  FairTestDetectorDigi* digi = new FairTestDetectorDigi(xPad, yPad, zPad, timestamp);
97  if (fTimeResolution > 0)
98  {
99  digi->SetTimeStampError(fTimeResolution / TMath::Sqrt(fTimeResolution));
100  }
101  else
102  {
103  digi->SetTimeStampError(0);
104  }
105 
106  digi->SetLink(FairLink("FairTestDetectorPoint", ipnt));
107 
108  Double_t timeOfFlight = point->GetTime();
109  Double_t eventTime = FairRootManager::Instance()->GetEventTime();
110 
111  fDataBuffer->FillNewData(digi, timeOfFlight + eventTime, digi->GetTimeStamp() + 10);
112  }
