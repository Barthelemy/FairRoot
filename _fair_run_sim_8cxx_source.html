141 void FairRunSim::Init()
142 {
143  /**Initialize the simulation session*/
144 
145  CheckFlukaExec();
146 
147 // fOutFile=fRootManager->OpenOutFile(fOutname);
148  fLogger->Info(MESSAGE_ORIGIN,"============== FairRunSim: Initialising simulation run ==============");
149 
150  FairGeoLoader* loader=new FairGeoLoader(fLoaderName->Data(), "Geo Loader");
151  FairGeoInterface* GeoInterFace=loader->getGeoInterface();
152  GeoInterFace->SetNoOfSets(ListOfModules->GetEntries());
153  GeoInterFace->setMediaFile(MatFname.Data());
154  GeoInterFace->readMedia();
155 
156  // gSystem->cd(flout.Data());
157 
158  fApp= new FairMCApplication("Fair","The Fair VMC App",ListOfModules, MatFname);
159  fApp->SetGenerator(fGen);
160 
161  // Add a Generated run ID to the FairRunTimeDb
162  FairRunIdGenerator genid;
163  // FairRuntimeDb *rtdb= GetRuntimeDb();
164  fRunId = genid.generateId();
165  fRtdb->addRun(fRunId);
166 
167  fFileHeader->SetRunId(fRunId);
168  /** Add Tasks to simulation if any*/
169  fApp->AddTask(fTask);
170 
171  FairBaseParSet* par=dynamic_cast<FairBaseParSet*>(fRtdb->getContainer("FairBaseParSet"));
172  par->SetDetList(GetListOfModules());
173  par->SetGen(GetPrimaryGenerator());
174  par->SetBeamMom(fBeamMom);
175 
176  FairGeoParSet* geopar=dynamic_cast<FairGeoParSet*>(fRtdb->getContainer("FairGeoParSet"));
177  geopar->SetGeometry(gGeoManager);
178 
179  // Set global Parameter Info
180 
181  if(fPythiaDecayer) {
182  fApp->SetPythiaDecayer(fPythiaDecayer);
183  if (fPythiaDecayerConfig) {
184  fApp->SetPythiaDecayerConfig(fPythiaDecayerConfig);
185  }
186  }
187  if(fUserDecay) {
188  fApp->SetUserDecay(fUserDecay);
189  if (fUserDecayConfig) {
190  fApp->SetUserDecayConfig(fUserDecayConfig);
191  }
192  }
193  // on/off visualisation
194  if( fStoreTraj ) {
195  fLogger->Info(MESSAGE_ORIGIN, "Create visualisation manager ");
196  new FairTrajFilter();
197  }
198  if(fRadLength) {
199  fApp->SetRadiationLengthReg(fRadLength);
200  }
201  if(fRadMap) {
202  fApp->SetRadiationMapReg(fRadMap);
203  }
204  if(fRadGrid) {
205  fApp->AddMeshList(fMeshList);
206  }
207 
208  if(fField) { fField->Init(); }
209  fApp->SetField(fField);
210  SetFieldContainer();
211 
212  TList* containerList=fRtdb->getListOfContainers();
213  TIter next(containerList);
214  FairParSet* cont;
215  TObjArray* ContList= new TObjArray();
216  while ((cont=dynamic_cast<FairParSet*>(next()))) {
217  ContList->Add(new TObjString(cont->GetName()));
218  }
219 
220  par->SetContListStr(ContList);
221  par->setChanged();
222  par->setInputVersion(fRunId,1);
223 
224  geopar->setChanged();
225  geopar->setInputVersion(fRunId,1);
226 
227 
228 
229  /**Set the configuration for MC engine*/
230  SetMCConfig();
231  fRootManager->WriteFileHeader(fFileHeader);
232 
233 }

242 void FairRunSim::CheckFlukaExec()
243 {
244  /** Private method for setting FLUKA simulation*/
245  TString work = getenv("VMCWORKDIR");
246  TString work_config=work+"/gconfig/";
247  TString config_dir= getenv("CONFIG_DIR");
248  if (!config_dir.EndsWith("/")) { config_dir+="/"; }
249  TString flout;
250  if(strcmp(GetName(),"TFluka") == 0 ) {
251  TString flexec="run_fluka.sh";
252  if (TString(gSystem->FindFile(config_dir.Data(),flexec)) != TString("")) {
253  fLogger->Info(MESSAGE_ORIGIN,"---User path for Configuration is used : %s ", config_dir.Data());
254  } else {
255  flexec=work_config+"run_fluka.sh";
256  }
257  gSystem->Exec(flexec.Data());
258  // More generic for different macro
259  // dir layout <D.B>
260  TString macrodir = getenv("PWD");
261  cout << macrodir << endl;
262  flout = macrodir + "/fluka_out" ;
263  gSystem->cd(flout.Data());
264  }

267 void FairRunSim::SetMCConfig()
268 {
269  /** Private method for setting simulation and/or Geane configuration and cuts*/
270 
271  TString work = getenv("VMCWORKDIR");
272  TString Lib_config= getenv("GEANT4VMC_MACRO_DIR");
273  TString work_config=work+"/gconfig/";
274  TString config_dir= getenv("CONFIG_DIR");
275  Bool_t AbsPath=kFALSE;
276  if (!config_dir.EndsWith("/")) { config_dir+="/"; }
277  if (!Lib_config.EndsWith("/")) { Lib_config+="/"; }
278 
279  TString LibMacro;
280  TString LibFunction;
281  TString ConfigMacro;
282  TString cuts=fUserCuts;
283  //----------------------------------------------Geant4 Config-----------------------------------------
284  if(strcmp(GetName(),"TGeant4") == 0 ) {
285  TString g4LibMacro="g4libs.C";
286  TString g4Macro;
287  if(fUserConfig.IsNull()) {
288  g4Macro="g4Config.C";
289  fUserConfig = g4Macro;
290  } else {
291  if (fUserConfig.Contains("/")) { AbsPath=kTRUE; }
292  g4Macro = fUserConfig;
293  fLogger->Info(MESSAGE_ORIGIN,"---------------User config is used : %s ",g4Macro.Data());
294  }
295  if (TString(gSystem->FindFile(config_dir.Data(),g4LibMacro)) != TString("")) { //be carfull after this call the string g4LibMacro is empty if not found!!!!
296  fLogger->Info(MESSAGE_ORIGIN,"---User path for Configuration (g4libs.C) is used : %s" ,config_dir.Data());
297  } else {
298  g4LibMacro=Lib_config+"g4libs.C";
299  }
300  LibMacro=g4LibMacro;
301  LibFunction="g4libs()";
302  if (!AbsPath && TString(gSystem->FindFile(config_dir.Data(),g4Macro)) != TString("")) {
303  fLogger->Info(MESSAGE_ORIGIN,"---User path for Configuration (g4Config.C) is used : %s" , config_dir.Data());
304  ConfigMacro=g4Macro;
305  } else {
306  if(AbsPath) { ConfigMacro = fUserConfig; }
307  else { ConfigMacro =work_config+fUserConfig; }
308  }
309  //----------------------------------------------Geant3 Config-----------------------------------------
310  } else if(strcmp(GetName(),"TGeant3") == 0 ) {
311  TString g3LibMacro="g3libs.C";
312  TString g3Macro="g3Config.C";
313  if(fUserConfig.IsNull()) {
314  g3Macro="g3Config.C";
315  fUserConfig = g3Macro;
316  fLogger->Info(MESSAGE_ORIGIN,"-------------- Standard Config is called ------------------------------------");
317  } else {
318  if (fUserConfig.Contains("/")) { AbsPath=kTRUE; }
319  g3Macro = fUserConfig;
320  fLogger->Info(MESSAGE_ORIGIN,"---------------User config is used : ", g3Macro.Data());
321  }
322  if (TString(gSystem->FindFile(config_dir.Data(),g3LibMacro)) != TString("")) {
323  fLogger->Info(MESSAGE_ORIGIN,"---User path for Configuration (g3libs.C) is used : %s " , config_dir.Data());
324  } else {
325  g3LibMacro=work_config+"g3libs.C";
326  }
327  LibMacro=g3LibMacro;
328  LibFunction="g3libs()";
329  if (!AbsPath && TString(gSystem->FindFile(config_dir.Data(),g3Macro)) != TString("")) {
330  fLogger->Info(MESSAGE_ORIGIN,"---User path for Configuration (g3Config.C) is used : %s", config_dir.Data());
331  ConfigMacro=g3Macro;
332  } else {
333  if(AbsPath) { ConfigMacro = fUserConfig; }
334  else { ConfigMacro =work_config+fUserConfig; }
335  }
336  //----------------------------------------------Fluka Config-----------------------------------------
337  } else if(strcmp(GetName(),"TFluka") == 0 ) {
338  TString flLibMacro="fllibs.C";
339  TString flMacro="flConfig.C";
340  if(fUserConfig.IsNull()) {
341  flMacro="flConfig.C";
342  fUserConfig=flMacro;
343  } else {
344  if (fUserConfig.Contains("/")) { AbsPath=kTRUE; }
345  flMacro = fUserConfig;
346  fLogger->Info(MESSAGE_ORIGIN,"---------------User config is used : %s " , flMacro.Data());
347  }
348  if (TString(gSystem->FindFile(config_dir.Data(), flLibMacro)) != TString("")) {
349  fLogger->Info(MESSAGE_ORIGIN,"---User path for Configuration (fllibs.C) is used : %s", config_dir.Data());
350  } else {
351  flLibMacro=work_config+"fllibs.C";
352  }
353  LibMacro=flLibMacro;
354  LibFunction="fllibs()";
355  if (!AbsPath && TString(gSystem->FindFile(config_dir.Data(),flMacro)) != TString("")) {
356  fLogger->Info(MESSAGE_ORIGIN, "---User path for Configuration (flConfig.C) is used : %s", config_dir.Data());
357  ConfigMacro=flMacro;
358  } else {
359  if(AbsPath) { ConfigMacro = fUserConfig; }
360  else { ConfigMacro =work_config+fUserConfig; }
361  }
362  }
363  //----------------------------------------------SetCuts------------------------------------------------
364  if (TString(gSystem->FindFile(config_dir.Data(),cuts)) != TString("")) {
365  fLogger->Info(MESSAGE_ORIGIN,"---User path for Cuts and Processes (SetCuts.C) is used : %s", config_dir.Data());
366  } else {
367  cuts =work_config+ fUserCuts;
368  }
369  //--------------------------------------Now load the Config and Cuts------------------------------------
370  gROOT->LoadMacro(LibMacro.Data());
371  gROOT->ProcessLine(LibFunction.Data());
372 
373  gROOT->LoadMacro(ConfigMacro.Data());
374  gROOT->ProcessLine("Config()");
375 
376  gROOT->LoadMacro(cuts);
377  gROOT->ProcessLine("SetCuts()");
378 
379  fApp->InitMC(ConfigMacro.Data(), cuts.Data());
380 }
