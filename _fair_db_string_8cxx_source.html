68 void FairUtilString::MakePrintable(const Char_t* in,
69  std::string& out)
70 {
71 
72  Bool_t hasSpecial = false;
73  Int_t index = 0;
74  while ( UChar_t c = in[index++] ) {
75  if ( c == '\\' || c == '\n' || c == '\t' || c == '\'' || c == '\"'
76  || c <= '\x08' || (c >= '\x0b' && c <= '\x1f' ) || c >= '\x7f' ) {
77  hasSpecial = true;
78  break;
79  }
80  }
81  if ( ! hasSpecial ) {
82  out += in;
83  return;
84  }
85  index = 0;
86  while ( UChar_t c = in[index++] ) {
87  // Skip really unprInt_table ones.
88  if ( c <= '\x08' || (c >= '\x0b' && c <= '\x1f' ) || c >= '\x7f' ) { continue; }
89  if ( c == '\\' || c == '\n' || c == '\t' || c == '\'' || c == '\"') {
90  switch ( c ) {
91  case '\\':
92  out += "\\\\";
93  break;
94  case '\n':
95  out += "\\n";
96  break;
97  case '\t':
98  out += "\\t";
99  break;
100  case '\'':
101  out += "\\\'";
102  break;
103  case '\"':
104  out += "\\\"";
105  break;
106  }
107  } else { out += c; }
108  }
109 }

111 void FairUtilString::StringTok(std::vector<std::string>& ls,
112  const std::string& str,
113  const std::string& tok)
114 {
115  const string::size_type S = str.size();
116  const string::size_type toksz = tok.size();
117  string::size_type i = 0;
118 
119  while (i < S) {
120  // eat leading whitespace
121  while ( (i<S) && (tok.find(str[i])<=toksz) ) {
122  ++i;
123  }
124  if (i == S) { return; }
125 
126  string::size_type j = i+1;
127  while ( (j<S) && !(tok.find(str[j])<=toksz) ) {
128  ++j;
129  }
130 
131  ls.push_back(str.substr(i,j-i));
132 
133  i = j+1;
134  }
135 }

199 Int_t FairUtilString::cmp_nocase(const std::string& s1, const std::string& s2)
200 {
201  std::string::const_iterator p1=s1.begin();
202  std::string::const_iterator p2=s2.begin();
203  while (p1!=s1.end() && p2!=s2.end()) {
204  if (toupper(*p1) != toupper(*p2)) {
205  return (toupper(*p1)<toupper(*p2)) ? -1 : 1;
206  }
207  ++p1;
208  ++p2;
209  }
210  return (s2.size()==s1.size()) ? 0 : (s1.size()<s2.size()) ? -1:1;
211 }
