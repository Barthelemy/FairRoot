67 InitStatus FairMCTracks::Init()
68 {
69  if(fVerbose>1) { cout<< "FairMCTracks::Init()" << endl; }
70  FairRootManager* fManager = FairRootManager::Instance();
71  fTrackList = (TClonesArray*)fManager->GetObject("GeoTracks");
72  if(fTrackList==0) {
73  cout << "FairMCPointDraw::Init() branch " << GetName() << " Not found! Task will be deactivated "<< endl;
74  SetActive(kFALSE);
75  }
76  if(fVerbose>2) { cout<< "FairMCTracks::Init() get track list" << fTrackList<< endl; }
77  if(fVerbose>2) { cout<< "FairMCTracks::Init() create propagator" << endl; }
78  fEventManager =FairEventManager::Instance();
79  if(fVerbose>2) { cout<< "FairMCTracks::Init() get instance of FairEventManager " << endl; }
80  fEvent = "Current Event";
81  MinEnergyLimit=fEventManager->GetEvtMinEnergy();
82  MaxEnergyLimit=fEventManager->GetEvtMaxEnergy();
83  PEnergy=0;
84  if(IsActive()) { return kSUCCESS; }
85  else { return kERROR; }

88 void FairMCTracks::Exec(Option_t* option)
89 {
90 
91  if (IsActive()) {
92 
93  if(fVerbose>1) { cout << " FairMCTracks::Exec "<< endl; }
94  TGeoTrack* tr;
95  const Double_t* point;
96 
97  Reset();
98 
99  for (Int_t i=0; i<fTrackList->GetEntriesFast(); i++) {
100  if(fVerbose>2) { cout << "FairMCTracks::Exec "<< i << endl; }
101  tr=(TGeoTrack*)fTrackList->At(i);
102  TParticle* P=(TParticle*)tr->GetParticle();
103  PEnergy=P->Energy();
104  MinEnergyLimit=TMath::Min(PEnergy,MinEnergyLimit) ;
105  MaxEnergyLimit=TMath::Max(PEnergy,MaxEnergyLimit) ;
106  if(fVerbose>2) { cout << "MinEnergyLimit " << MinEnergyLimit << " MaxEnergyLimit " << MaxEnergyLimit << endl; }
107  if (fEventManager->IsPriOnly() && P->GetMother(0)>-1) { continue; }
108  if(fEventManager->GetCurrentPDG()!=0 && fEventManager->GetCurrentPDG()!= tr->GetPDG()) { continue; }
109  if(fVerbose>2) { cout << "PEnergy " << PEnergy << " Min " << fEventManager->GetMinEnergy() << " Max " << fEventManager->GetMaxEnergy() <<endl; }
110  if( (PEnergy<fEventManager->GetMinEnergy()) || (PEnergy >fEventManager->GetMaxEnergy())) { continue; }
111 
112  Int_t Np=tr->GetNpoints();
113  fTrList= GetTrGroup(P);
114  TEveTrack* track= new TEveTrack(P, tr->GetPDG(), fTrPr);
115  track->SetLineColor(fEventManager->Color(tr->GetPDG()));
116  for (Int_t n=0; n<Np; n++) {
117  point=tr->GetPoint(n);
118  track->SetPoint(n,point[0],point[1],point[2]);
119  TEveVector pos= TEveVector(point[0], point[1],point[2]);
120  TEvePathMark* path = new TEvePathMark();
121  path->fV=pos ;
122  path->fTime= point[3];
123  if(n==0) {
124  TEveVector Mom= TEveVector(P->Px(), P->Py(),P->Pz());
125  path->fP=Mom;
126  }
127  if(fVerbose>3) { cout << "Path marker added " << path << endl; }
128 
129  track->AddPathMark(*path);
130 
131  if(fVerbose>3) { cout << "Path marker added " << path << endl; }
132 
133  delete path;
134  }
135  fTrList->AddElement(track);
136  if(fVerbose>3) { cout << "track added " << track->GetName() << endl; }
137 
138  }
139  for (Int_t i=0; i<fEveTrList->GetEntriesFast(); i++) {
140  // TEveTrackList *TrListIn=( TEveTrackList *) fEveTrList->At(i);
141  //TrListIn->FindMomentumLimits(TrListIn, kFALSE);
142  }
143  fEventManager->SetEvtMaxEnergy(MaxEnergyLimit);
144  fEventManager->SetEvtMinEnergy(MinEnergyLimit);
145  gEve->Redraw3D(kFALSE);
146  }
147 }

173 TEveTrackList* FairMCTracks::GetTrGroup(TParticle* P)
174 {
175 
176  fTrList=0;
177  for (Int_t i=0; i<fEveTrList->GetEntriesFast(); i++) {
178  TEveTrackList* TrListIn=( TEveTrackList*) fEveTrList->At(i);
179  if ( strcmp(TrListIn->GetName(),P->GetName())==0 ) {
180  fTrList= TrListIn;
181  break;
182  }
183  }
184  if(fTrList ==0) {
185  fTrPr=new TEveTrackPropagator();
186  fTrList= new TEveTrackList(P->GetName(),fTrPr );
187  fTrList->SetMainColor(fEventManager->Color(P->GetPdgCode()));
188  fEveTrList->Add(fTrList);
189  gEve->AddElement( fTrList ,fEventManager );
190  fTrList->SetRnrLine(kTRUE);
191  }
192  return fTrList;
193 }
