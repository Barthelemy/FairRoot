140 Int_t FairTutorialDet4GeoHandler::CurrentVolOffID(Int_t off, Int_t& copy) const
141 {
142  if (fIsSimulation) {
143  return gMC->CurrentVolOffID(off, copy);
144  } else {
145  //
146  // Return the current volume "off" upward in the geometrical tree
147  // ID and copy number
148  //
149  if (off<0 || off>gGeoManager->GetLevel()) { return 0; }
150  if (off==0) { return CurrentVolID(copy); }
151  TGeoNode* node = gGeoManager->GetMother(off);
152  if (!node) { return 0; }
153  copy = node->GetNumber();
154  return node->GetVolume()->GetNumber();
155  }
156 }

192 void FairTutorialDet4GeoHandler::NavigateTo(TString volName)
193 {
194  if (fIsSimulation) {
195  LOG(FATAL)<<"This methode is not supported in simulation mode"<<FairLogger::endl;
196  } else {
197  gGeoManager->cd(volName.Data());
198  fGeoPathHash = volName.Hash();
199  fCurrentVolume = gGeoManager->GetCurrentVolume();
200  fVolumeShape = (TGeoBBox*)fCurrentVolume->GetShape();
201  Double_t local[3] = {0., 0., 0.}; // Local centre of volume
202  gGeoManager->LocalToMaster(local, fGlobal);
203  LOG(DEBUG2)<<"Pos: "<<fGlobal[0]<<" , "<<fGlobal[1]<<" , "<<fGlobal[2]<<FairLogger::endl;
204 // fGlobalMatrix = gGeoManager->GetCurrentMatrix();
205  }
206 }
